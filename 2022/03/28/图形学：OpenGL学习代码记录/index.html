DDA算法

利用微分方程来确定x，y方向上增长，每次根据delt(x)和delt(y)来逐次花像素点。

void DDALine(int x0, int y0, int x1, int y1, int color) {
    int dx = x1 - x0, dy = y1 - y0;
    float x_increase, y_increase, steps, x = x0, y = y0;
    // &#x659C;&#x7387;&#x5927;&#x4E8E;1&#x5219;&#x6B65;&#x957F;&#x6570;&#x4E3A;x&#x5DEE;&#xFF0C; &#x659C;&#x7387;&#x5C0F;&#x4E8E;1&#x5219;&#x6B65;&#x957F;&#x6570;&#x4E3A;y&#x5DEE;
    if(abs(dx) > abs(dy)) steps = abs(dx);
    else steps = abs(dy);
    x_increase = (float)(dx) / steps;
    y_increase = (float)(dy) / steps;
    for(int i = 0; i < steps; i ++) {
        putpixel(round(x), round(y), color);
        x += x_increase; y += y_increase;
    }
}

Bresenham算法

假设斜率在0到1之间

void BresenhamLine(int x0, int y0, int x1, int y1, int color) {
    int dx, dy, h, x, y;
    dx = abs(x1 - x0); dy = abs(y1 - y0);
    h = 2 * dy - dx; x = x0; y = y0;
    putpixel(x, y, color);
    while(x < x1) {
        if(h < 0) h += 2 * dy;
        else h += 2 * (dy - dx), ++ y;
        putpixel(x, y, color);
        ++ x;
    }
}

圆的生成

中点Bresenham算法

根据圆的对称性，只画1/8圆弧即可，从(0, R)到(R / sqrt(2), R / sqrt(2))。

与直线的Bresenham算法类似原理，若当前像素点(xi, yi)更加接近圆，那么就需要找到下一个更加接近圆的点。*、

void BresenhameCircle(int x0, int y0, double radius, int color) {
    int x, y, h;
    x = 0; y = int(radius); h = 1 - int(radius);
    CirPot(x0, y0, x, y, color);
    while(x < y) {
        if(h < 0) h += 2 * x + 3;
        else h += 2 * (x - y) + 5, -- y;
        ++ x;
        CirPot(x0, y0, x, y, color);
    }
}

// &#x7ED8;&#x5236;8&#x4E2A;&#x5BF9;&#x79F0;&#x70B9;
void CirPot(int x0, int y0, int x, int y, int color) {
    putpixel(x0 + x, y0 + y, color);
    putpixel(x0 + x, y0 - y, color);
    putpixel(x0 - x, y0 + y, color);
    putpixel(x0 - x, y0 - y, color);
    putpixel(x0 + y, y0 + x, color);
    putpixel(x0 + y, y0 - x, color);
    putpixel(x0 - y, y0 + x, color);
    putpixel(x0 - y, y0 - x, color);
}

#include <GL/glut.h>
#include <stdio.h>
#include <math.h>

const float PI = acos(-1.0);
int n = 6, R = 10;  // &#x591A;&#x8FB9;&#x5F62;&#x8FB9;&#x6570;&#x548C;&#x534A;&#x5F84;

float theta = 0.0;  // &#x65CB;&#x8F6C;&#x521D;&#x59CB;&#x89D2;&#x5EA6;
void Keyboard(unsigned char key, int x, int y);  // &#x952E;&#x76D8;&#x54CD;&#x5E94;&#x51FD;&#x6570;
void display(void);  // &#x663E;&#x793A;&#x56DE;&#x8C03;&#x51FD;&#x6570;
void reshape(int w, int h);  // &#x7A97;&#x53E3;&#x6539;&#x53D8;&#x56DE;&#x8C03;&#x51FD;&#x6570;
void myidle();  // &#x95F2;&#x7F6E;&#x51FD;&#x6570;

int main(int argc, char** argv)
{
    glutInit(&argc, argv);
    glutInitWindowSize(700, 700);
    glutInitDisplayMode(GLUT_DOUBLE | GLUT_RGB);
    glutCreateWindow("a rotating square");
    glutDisplayFunc(display);
    glutReshapeFunc(reshape);
    glutIdleFunc(myidle);
    glutMainLoop();

    return 0;
}

void display(void) {
    glClear(GL_COLOR_BUFFER_BIT);
    glColor3f(1.0, 0, 0);  // red
    glBegin(GL_POLYGON);
        for(int i = 0; i < n; i ++)
            glVertex2f(R*cos(theta + i*2*PI/n), R*sin(theta + i*2*PI/n));
    glEnd();
    glutSwapBuffers();
}

void myidle() {
    theta += 1.0;
    if(theta >= 2*PI) theta -= 2*PI;
    glutPostRedisplay();  // &#x91CD;&#x753B;&#xFF0C;&#x76F8;&#x5F53;&#x4E8E;&#x91CD;&#x65B0;&#x8C03;&#x7528;&#x4E86;display
}

void reshape(GLsizei w, GLsizei h) {
    glMatrixMode(GL_PROJECTION);
    glLoadIdentity();
    gluOrtho2D(-1.5*R*w/h, 1.5*R*w/h, -1.5*R, 1.5*R);
    glViewport(0, 0, w, h);
    glMatrixMode(GL_MODELVIEW);
}
