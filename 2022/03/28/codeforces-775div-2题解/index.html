A Game

题解

找到最左边和最右边的0即可

代码

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int n, a[110];

int main()
{
    #ifdef ONLINE_JUDGE
    #else
        freopen("data.in", "r", stdin);
    #endif
    int T;
    scanf("%d\n", &T);
    while(T--) {
        scanf("%d", &n);
        for(int i = 1; i <= n; i ++) scanf("%d", a + i);
        int i = 1, j = n;
        a[0] = a[n + 1] = 0;
        while(a[i]) ++ i;
        while(a[j]) -- j;
        if(i == n + 1) puts("0");
        else printf("%d\n", j - i + 2);
    }
    return 0;
}

B Game of Ball Passing

题解

先排序，然后取数最大的那个当作主链，然后让其他数往里面插，如果其他数都不够插入（即所有数插入之后，数最大的那个人还是有连续的情况，（根据题意，自己不能传给自己）），那么剩下的数就各自单独成一条链。

如果其他数加起来足够插入的话说明，最大的那个数不够大，插法是先把主链插满（即没有自己传给自己的情况），然后后面小的数一定能够插进主链中。

代码

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int n;
int a[100010];

int main()
{
    #ifdef ONLINE_JUDGE
    #else
        freopen("data.in", "r", stdin);
    #endif
    int T;
    scanf("%d\n", &T);
    while(T--) {
        scanf("%d", &n);
        int ok = 0;
        for(int i = 1; i <= n; i ++) {
            scanf("%d", a + i);
            if(a[i]) ok = 1;
        }
        if(!ok) {
            puts("0");
            continue;
        }
        sort(a + 1, a + n + 1);
        ll mainn = a[n], sum = 0;
        for(int i = n - 1; i >= 1; i --) sum += a[i];
        if(sum >= mainn - 1) puts("1");
        else printf("%lld\n", mainn - sum);
    }
    return 0;
}

C Weird Sum

题解

对于两个点，其距离为|xi - xj| + |yi - yj|。

由于绝对值的存在，而且还是二维了，所以不能够通过排序然后前缀和的办法快速计算，但是可以发现，x上的贡献和y上的贡献是不相互影响的，那么就循环两次，一次x为主循环，一次y为主循环，然后分别计算贡献即可。

代码

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int n, m;
vector<int> x[100010], y[100010];
ll sumx[100010], sumy[100010];

int main()
{
    #ifdef ONLINE_JUDGE
    #else
        freopen("data.in", "r", stdin);
    #endif
    scanf("%d%d", &n, &m);
    vector<vector<int>> G;
    G.resize(n + 1);
    for(int i = 1; i <= n; i ++) G[i].resize(m + 1);
    for(int i = 1; i <= n; i ++) for(int j = 1; j <= m; j ++) scanf("%d", &G[i][j]);

    ll res = 0;
    for(int i = 1; i <= n; i ++) for(int j = 1; j <= m; j ++) {
        int c = G[i][j];
        if(x[c].size()) res += 1ll * i * x[c].size() - sumx[c];
        x[c].push_back(i);
        sumx[c] += i;
    }

    for(int j = 1; j <= m; j ++) for(int i = 1; i <= n; i ++) {
        int c = G[i][j];
        if(y[c].size()) res += 1ll * j * y[c].size() - sumy[c];
        y[c].push_back(j);
        sumy[c] += j;
    }

    printf("%lld\n", res);

    return 0;
}
