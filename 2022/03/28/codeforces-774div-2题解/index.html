A Square Counting

题解

答案是唯一的，再根据样例就能猜出来。

由于(n+1)(n-1) = n * n - 1 < n * n，所以要尽量多的n*n，否则剩下的凑不够了。

答案为 s / n / n。

代码

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main()
{
    #ifdef ONLINE_JUDGE
    #else
        freopen("data.in", "r", stdin);
    #endif
    int T;
    scanf("%d\n", &T);
    while(T--) {
        ll n, s; scanf("%lld%lld", &n, &s);
        n *= n;
        printf("%lld\n", s / n);
    }
    return 0;
}

B Quality vs Quantity

题解

贪心，尽量找最大的染成红色，为了防止红的个数大于等于蓝色的个数，每次从大到小选一个红，就得从小到大选一个蓝，保证count(蓝)>count(红)，如果发现sumr > sumb则YES。

代码

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int n, a[200010];

int main()
{
    #ifdef ONLINE_JUDGE
    #else
        freopen("data.in", "r", stdin);
    #endif
    int T;
    scanf("%d\n", &T);
    while(T--) {
        scanf("%d", &n);
        for(int i = 1; i <= n; i ++) scanf("%d", a + i);
        sort(a + 1, a + n + 1);
        ll sumb = 0, sumr = 0; a[n + 1] = 0;
        int i = 1, j = n + 1, ok = 0;
        while(i < j) {
            sumb += a[i++];
            sumr += a[j--];
            if(sumr > sumb) {
                ok = 1;
                puts("YES");
                break;
            }
        }
        if(!ok) puts("NO");
    }
    return 0;
}

C Factorials and Powers of Two

题解

给一个n都能够找到不同的数表示（由二进制可以想到），那么不同的阶乘数相加可以代替一系列的2的n次幂数，那么枚举所有阶乘数相加的情况，处理出来。

powerful bumber分两种：一种为阶乘数，一种为二进制数。

那么对于每一个n，可以由阶乘数和二进制数组成，对于每一个处理出来的阶乘数，n减去其后剩下就全用二进制数表示，枚举每一种情况，取最小即可。

代码

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

vector<ll> v, w, wc;
void init() {
    ll pp = 6;
    for(ll i = 4; pp <= 1e12; i ++) {
        v.push_back(pp);
        pp *= i;
    }

    for(ll i = 0; i < (1 << 12); i ++) {
        ll sum = 0;
        for(int j = 0; j < 12; j ++)
            if(i >> j & 1)
                sum += v[j];
        w.push_back(sum), wc.push_back(__builtin_popcountll(i));
    }
}

int main()
{
    #ifdef ONLINE_JUDGE
    #else
        freopen("data.in", "r", stdin);
    #endif
    init();
    int T;
    scanf("%d\n", &T);
    while(T--) {
        ll n; scanf("%lld", &n);
        ll res = __builtin_popcountll(n);
        // cout << "init res is " << res << endl;
        for(int i = 0; i < w.size(); i ++)
            if(w[i] <= n)
                res = min((ll)res, __builtin_popcountll(n - w[i]) + wc[i]);
        printf("%lld\n", res);
    }
    return 0;
}
