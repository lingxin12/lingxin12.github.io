介绍

由于某些条件的限制，并没有使用games202提供的框架来完成作业，思考再三，决定使用OpenGL和GLSL来实现games202的作业，由于不知道后面的作业到底还会有什么限制，至少目前来说OpenGL还是可以实现shadow mapping的，所以使用OpenGL来实现相关内容。

在作业中其实没有提到的一点（我也不知道有没有提到，WebGL的代码看了大概，没仔细看）是，对于不同的光源其shadow map其实实现方式是不同的，对于平行光，只需要朝着需要生成阴影的方向生成shadow map即可，但是对于点光源，甚至是聚光灯其shadow map的生成其实是不一样的，比如点光源就需要任意一个方向都要能够查询到shadow map。

由于刚开始没有考虑到这一点，所以代码中，虽然使用的是平行光，但是phone模型还是添加上的光线衰减之类的，但是这并不影响阴影生成。这里会结合learnopengl的阴影生成，会分别考虑使用平行光和点光源来实现阴影。

下面现在unity中看一下平行光和点光源它们的阴影区别：

上图中，比较淡一点的阴影是点光源生成的阴影，而暗一点的是平行光生成的阴影。

vertex.glsl

#version 330 core
layout (location = 0) in vec3 aPos;
layout (location = 1) in vec3 aNormal;
layout (location = 2) in vec2 aTexCoords;

out vec3 fragCoords;
out vec3 Normal;
out vec2 texCoords;

uniform mat4 projection;
uniform mat4 view;
uniform mat4 model;

void main()
{
    fragCoords = vec3(model * vec4(aPos, 1.0));
    Normal = mat3(inverse(transpose(model))) * aNormal;
    texCoords = aTexCoords;
    gl_Position = projection * view * model * vec4(aPos, 1.0);
}

#version 330 core
out vec4 FragColor;

in vec3 fragCoords;
in vec3 Normal;
in vec2 texCoords;

uniform vec3 viewPos;
uniform vec3 lightStrength;
uniform vec3 lightPos;
uniform bool is_floor;

uniform float linear;
uniform float quadratic;

const float smoothStrength = 4;
const float ambientStrength = 0.1;

/*******************************************************************/
/*
*
* shadow mapping: 2-pass, PCF, PCSS
*
*/

uniform mat4 depthProjection;
uniform mat4 depthView;

uniform int shadowMapChoose;
uniform sampler2D texture_depth;

const float shadow_width = 1024.0;
const float shadow_height = 1024.0;

// float bias = 0.005;
float bias = max(0.01 * (1.0 - dot(Normal, normalize(lightPos - fragCoords))), 0.005);

float TwoPass()
{
    return 1.0;
}

float PCF()
{
    return 1.0;
}

float PCSS()
{
    return 1.0;
}

float ShadowMapping()
{
    if(shadowMapChoose == 0) return 1.0;
    else if(shadowMapChoose == 1) return TwoPass();
    else if(shadowMapChoose == 2) return PCF();
    else if(shadowMapChoose == 3) return PCSS();
    return 1.0;
}

/*******************************************************************/


vec4 Phone(bool is_texture, vec3 color, float smoothFactor)
{
    vec4 objectColor;
    if(is_texture)
        objectColor = texture(texture_diffuse1, texCoords);
    else
        objectColor = vec4(color, 1.0);

    vec3 ambient = ambientStrength * lightStrength;

    vec3 l_in = normalize(lightPos - fragCoords);
    vec3 N = normalize(Normal);
    vec3 diffuse = max(dot(l_in, N), 0) * lightStrength;
    
    vec3 viewVec = normalize(viewPos - fragCoords);
    vec3 h = normalize(viewPos + l_in);
    vec3 v_out = normalize(reflect(-l_in, N));
    vec3 specular = pow(max(dot(h, N), 0), smoothStrength * smoothFactor) * lightStrength;
    
    if(dot(viewVec, N) <= 0) diffuse = specular = vec3(0, 0, 0);

    float dis = length(lightPos - fragCoords);
    float attenuation = 1.0 / (constant + linear * dis + quadratic * (dis * dis));

    vec3 result = (ambient + diffuse + specular) * attenuation * vec3(objectColor);
    return  vec4(result, 1.0);
}

vec4 cal_floor()
{
    // return vec4(0.4, 0.4, 0.4, 1.0);
    return ShadowMapping() * Phone(false, vec3(0.5, 0.5, 0.5), 1);
}

vec4 cal_marry()
{
    // return texture(texture_diffuse1, texCoords);
    return ShadowMapping() * Phone(true, vec3(0, 0, 0), 8);
}

void main()
{
    if(is_floor)
        FragColor = cal_floor();
    else
        FragColor = cal_marry();
}

shadow mapping（平行光）

2-PASS

其实，一切事物都难于开始（废话）。

这一步主要的难点有两个：

  1. 生成shadow map。
  2. 对于每一个shading point，在shadow map找到对应的位置查询深度。

首先是第一点，这需要我们把光源当成相机，从光源处看向场景，然后进行第一次渲染，这次渲染我们只需要得到深度信息。该处的难点在于，变换矩阵到底该如何设置（点光源和平行光是有一点点不同的）？深度信息应该存在一个多大的shadow map上？这部分的代码其实不算特别难，但是细节还挺多的，容易在一些小地方出错。

其次是第二点，处理好上面那一步的矩阵之后，其实这一部分没有多难了，但是还是需要理解一下各个坐标空间的变换，特别是裁剪空间（clip space）和视口空间（viewport space）的变换。

使用帧缓存获得深度贴图

想要生成shadow map，需要用到OpenGL中的帧缓存技术，然后链接到一个只用来存储深度的纹理。

unsigned int framebuffer;
glGenFramebuffers(1, &framebuffer);
glBindFramebuffer(GL_FRAMEBUFFER, framebuffer);
unsigned int depth_map;
glGenTextures(1, &depth_map);
glBindTexture(GL_TEXTURE_2D, depth_map);
glTexImage2D(GL_TEXTURE_2D, 0, GL_DEPTH_COMPONENT, SHADOW_WIDTH, SHADOW_HEIGHT, 0, GL_DEPTH_COMPONENT, GL_FLOAT, NULL);
glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
glFramebufferTexture2D(GL_FRAMEBUFFER, GL_DEPTH_ATTACHMENT, GL_TEXTURE_2D, depth_map, 0);
glDrawBuffer(GL_NONE);
glReadBuffer(GL_NONE);
if (glCheckFramebufferStatus(GL_FRAMEBUFFER) != GL_FRAMEBUFFER_COMPLETE)
    cout << "ERROR::FRAMEBUFFER:: Framebuffer is not complete!" << endl;
glBindFramebuffer(GL_FRAMEBUFFER, 0);

这里先生成一个framebuffer用来做帧缓存，然后使用glFramebufferTexture2D()函数来添加纹理附件，而glDrawBuffer()和glReadBuffer()主要用来设置不需要写入颜色或者读出颜色缓存（具体可以参考learnopengl），因为我们只需要深度信息。

然后使用一个shader专门用来生成深度贴图：

floor.Draw(depthShader);
model = glm::translate(model, glm::vec3(2.0f, 0, 0));
depthShader.setMat4("model", model);
mary.Draw(depthShader);
model = glm::translate(glm::mat4(1.0f), glm::vec3(-2.0f, 0, 0));
model = glm::scale(model, glm::vec3(0.4f, 0.4f, 0.4f));
depthShader.setMat4("model", model);
mary.Draw(depthShader);

model矩阵不必说，按照每个物体其在世界空间的坐标进行相应的变换即可，这里需要注意的是projection和view矩阵的生成。

glm::mat4 depthProjection = glm::ortho(-10.0f, 10.0f, -10.0f, 10.0f, 0.1f, 10.0f);
glm::mat4 depthView = glm::lookAt(lightPos, glm::vec3(0, 0, 0), glm::vec3(0, 1.0f, 0));

首先是view矩阵，这里使用lookAt()函数来生成，这里需要传入三个参数，分别是视野位置、目标方向、UP方向通过这三个参数就能够自动计算出相应的view矩阵。

这里比较需要注意的是projection矩阵是使用正交矩阵来生成的，而不是使用透视矩阵，因为对于平行光来说，光是平行的，不是从一个点发射出来的，如果使用透视矩阵会得到一个错误的结果（阴影会变大，可以参考上面unity那张图）。

同时我们可以发现，深度贴图的大小是1024 * 1024，其实对于帧缓存来说，我们的帧缓存是不会直接显示在屏幕上的，一般是经过一些操作的之后，帧缓存会作为纹理贴在整个屏幕上。所以生成深度贴图是不限制大小精度的（毕竟该深度贴图只用来做shadow对画面的显示没有直接影响），所以最好用一张长宽一致的贴图，这样方便做视口变换来找到纹理坐标。这样避免了在[0, 1]的范围中，出现每个纹理像素不是正方形的情况。

生成深度贴图的shader比较简单，只需要做简单的坐标变换即可，甚至只需要一个空片段着色器即可，因为我们不需要写入任何颜色。

2-PASS生成阴影的片段着色器

最后在fragment中实现twoPass()函数。

// float bias = 0.005;
float bias = max(0.01 * (1.0 - dot(Normal, normalize(lightPos - fragCoords))), 0.005);

vec3 GetDepthTextureCoord()
{
    vec4 pos = depthProjection * depthView * vec4(fragCoords, 1.0);
    pos = pos / pos.w;
    pos = pos * 0.5 + 0.5;
    return vec3(pos.x, pos.y, pos.z);
}

float TwoPass()
{
    // return vec4(1.0, 1.0, 1.0, 1.0);

    vec3 pos = GetDepthTextureCoord();
    float lightSpaceDepth = texture(texture_depth, pos.xy).r;
    float currentDepth = pos.z;
    return currentDepth - bias > lightSpaceDepth ? 0.1 : 1.0;
}

这里需要一个GetDepthTextureCoord()函数来计算每一个世界坐标下的shading point，对应到光源空间下的坐标值，然后比较对应的深度即可。

注意到，这里在比较深度的时候，当前点的深度减去了一个bias（偏移量），这是防止失真而加的。

如果不使用bias就会造成以下情况：

如果使用固定值bias = 0.005，以下是正常实现的情况：

使用固定的bias，其实会导致一些错误的问题，比如bias过大或者过小，比如裙底一些部分应该是暗的，却因为bias的存在变亮了：

这时候，可以使用动态的bias，根据法向量和光源向量的夹角来动态改变bias的大小。

float bias = max(0.01 * (1.0 - dot(Normal, normalize(lightPos - fragCoords))), 0.005);

个人感觉，其实这么设置还是会有一些不合理的地方，但是大体上是正确的，至少，正常人看到的地方基本上不会出现什么大问题（正常人不会往裙底看），因为bias主要只是为了解决表面（比如地面）失真的问题。

PCF

其实有了2-pass，一切都变得简单了起来，这里只需要设置一个filter_size，然后在一个范围内进行深度比较，最终决定阴影的软硬程度。

float bias = max(0.005 * (1.0 - dot(Normal, normalize(lightPos - fragCoords))), 0.001);
float PCF()
{
    vec3 pos = GetDepthTextureCoord();
    int visible_point = 0;
    for(int i = -filter_size; i <= filter_size; i ++)
        for(int j = -filter_size; j <= filter_size; j ++) {
            float lightSpaceDepth = texture(texture_depth, 
                          vec2(pos.x + i/shadow_width, pos.y +j/shadow_height)).r;
            float currentDepth = pos.z;
            if(currentDepth - bias <= lightSpaceDepth)
                ++ visible_point;
        }
    float shadow_value = 0.1 + 
   (float(visible_point) / float((filter_size * 2 + 1) * (filter_size * 2 + 1))) * 0.9;
    return shadow_value;
}

这里控制i和j向量，分别在[-filter_size, filter_size]的范围内，进行深度查询，然后与当前深度比较即可。最后的shadow_value是根据一个比例来计算出来的，同时设置了一个最小值0.1，表示阴影再按也有轻微的环境光所照射。

PCSS

在一次偶然的搜集资料的过程中，看到这么一句有关于PCSS的描述，其实非常直接、暴力、简单。

其实PCSS和PCF没什么区别，其区别就在于它能够动态地决定filter_size的大小，更加真实地模拟了现实情况。

有关PCSS的细节个人比较有感悟的几个地方：

  1. 关于filter_size的生成，需要用到平均block值，刚开始我一直认为是整块filter的平均，但是实际上是 block的平均，也就是遮挡了当前像素点的那些点才算，没遮挡的不算block。
  2. 关于filter_size的计算，有公式(d_receiver - d_blocker)*w_light/d_blocker，可以转换成(d_receiver/d_blocker - 1)*w_light，同时还明确d_reveiver>=d_blocker，这样子可以直观地感受到，如果d_receiver距离d_blocker越来越远，可见上述的式子就会越来越大，也就是filter_size越来越大，阴影越来越软，非常符合实际情况。同时可以把w_light当作一个放大因子，其越大阴影越软，也很符合其物理含义和实际情况。

实现代码

const float w_light = 10;
int block_filter_size = 10;
int filter_size = 4;

float PCSS()
{
    // step1: cal the average of block depth
    vec3 pos = GetDepthTextureCoord();
    int block_num = 0;
    float blocks_depth = 0;
    float now_block_depth = pos.z;
    for(int i = -block_filter_size; i <= block_filter_size; i ++) {
        for(int j = -block_filter_size; j <= block_filter_size; j ++) {
            float depth = texture(texture_depth, 
                          vec2(pos.x + i/shadow_width, pos.y + j/shadow_height)).r;
            if(depth <= now_block_depth - bias) {
                ++ block_num;
                blocks_depth += depth;
            }
        }
    }
    if(block_num != 0) blocks_depth /= block_num;
    
    // step2: cal the filter_size
    if(block_num == 0) filter_size = 0;
    else {
        filter_size = int((now_block_depth - blocks_depth) * w_light / blocks_depth);
        filter_size = min(filter_size, 20);
    }

    // step3: PCF
    return PCF();
}

第一步其实和PCF基本上没有什么区别，只不过是用的一个固定的block_filter_size来来采样深度贴图罢了，同时在第二步中，需要对一些特殊值进行处理，不处理也可以，只不过处理了可以降低程序崩溃的风险，该检测的时候还是要检测的。

小总结

其实通过全面地观察，可以发现，其还有很多不足的地方，比如上图就非常明显，它非常取决于filter_size的大小，还有bias的优化，容易出现阴影与人物分离的情况。

VSSMの败北

其实刚开始实现PCSS的时候，并不是使用的这种直接循环的比较粗鲁的方式，因为这种大循环的逻辑计算其实在着色器中直接算会导致一个很消耗资源的问题，上述的程序只要把block_filter_size和w_light设置得稍微高一些，就会导致画面掉帧的情况，这导致非常吃GPU的算力，这种逻辑上的计算最好还是交给CPU来算比较好。

所以刚开始直接考虑的是用vssm来优化一下pcss。

但是还是有难度的，这主要集中在以下几个点：

  1. 如何传递前缀和数组和二次前缀和数组，这里是可以考虑纹理传递，但是在传递过程中出现了一些值错误的情况，目前还没有解决。
  2. glReadPixel函数获取帧缓冲慢。
  3. 考虑过全部使用帧缓冲来进行数值传递，但是过程复杂。
  4. 这其中涉及到了大量运算，在着色器中很难进行调试。

这里给出初版的片段着色器代码，其中全部是照着vssm对应的公式来进行计算的。

float PCSS()
{
    // step1: cal the average depth of block
    vec3 pos = GetDepthTextureCoord();
    int l1 = int(pos.x * shadow_width) - block_filter_size - 1,
        r1 = int(pos.y * shadow_height) - block_filter_size - 1,
        l2 = int(pos.x * shadow_width) + block_filter_size,
        r2 = int(pos.y * shadow_height) + block_filter_size;
    float avg = (texelFetch(texture_shadow_map, ivec2(l2, r2), 0).r
                - texelFetch(texture_shadow_map, ivec2(l2, r1), 0).r
                - texelFetch(texture_shadow_map, ivec2(l1, r2), 0).r
                + texelFetch(texture_shadow_map, ivec2(l1, r1), 0).r) / float(block_filter_size) / float(block_filter_size);
    float square_avg = (texelFetch(texture_square_shadow_map, ivec2(l2, r2), 0).r
                       - texelFetch(texture_square_shadow_map, ivec2(l2, r1), 0).r
                       - texelFetch(texture_square_shadow_map, ivec2(l1, r2), 0).r
                       + texelFetch(texture_square_shadow_map, ivec2(l1, r1), 0).r) / float(block_filter_size) / float(block_filter_size);
    float variance = square_avg - avg * avg;
    float Pxt = variance / (variance + (pos.z - avg) * (pos.z - avg));
    float z_occ = (avg - Pxt * pos.z) / (1 - Pxt);
    
    // step2: cal the filter_size
    float w_penumbra = (pos.z - z_occ) * w_light / z_occ;
    filter_size = max(int(w_penumbra), 0);

    // step3: PCF
    l1 = int(pos.x * shadow_width) - filter_size - 1,
    r1 = int(pos.y * shadow_height) - filter_size - 1,
    l2 = int(pos.x * shadow_width) + filter_size,
    r2 = int(pos.y * shadow_height) + filter_size;
    avg = (texelFetch(texture_shadow_map, ivec2(l2, r2), 0).r
          - texelFetch(texture_shadow_map, ivec2(l2, r1), 0).r
          - texelFetch(texture_shadow_map, ivec2(l1, r2), 0).r
          + texelFetch(texture_shadow_map, ivec2(l1, r1), 0).r) / float(filter_size) / float(filter_size);
    square_avg = (texelFetch(texture_square_shadow_map, ivec2(l2, r2), 0).r
                 - texelFetch(texture_square_shadow_map, ivec2(l2, r1), 0).r
                 - texelFetch(texture_square_shadow_map, ivec2(l1, r2), 0).r
                 + texelFetch(texture_square_shadow_map, ivec2(l1, r1), 0).r) / float(filter_size) / float(filter_size);
    variance = square_avg - avg * avg;
    Pxt = variance / (variance + (pos.z - avg) * (pos.z - avg));

    float shadow_value = 0.1 + Pxt * 0.9;
    return shadow_value;
}

这里主要使用texelFetch来获取纹理的真实值，但是使用颜色调试的过程中发现传递过来的数组的值完全不对，这造成了很多麻烦，以后再考虑一下优化，或者考虑再unity中实现。
