uva12219表达式数类型的例题

2020-03-22- 2021-03-07
题解 > 紫书-算法ACM
题目大意请参见紫书P354。

分析在书上有。

具体见代码：

#include<bits/stdc++.h>
using namespace std;
const int maxn = 5e4 + 3;

//&#x4E0B;&#x9762;&#x8FD9;&#x4E2A;&#x7ED3;&#x6784;&#x4F53;&#x7528;&#x4E8E;&#x5B58;&#x50A8;&#x4E00;&#x4E2A;&#x8282;&#x70B9;&#xFF0C;&#x6839;&#x636E;&#x4E66;&#x4E2D;&#x7684;&#x63CF;&#x8FF0;&#xFF0C;&#x6211;&#x4EEC;&#x53EF;&#x4EE5;&#x5C06;&#x4E00;&#x9897;&#x5B50;&#x6811;Node(s,l,r)&#x6620;&#x5C04;&#x6210;&#x7F16;&#x53F7;1&#xFF0C;2&#xFF0C;3....
//&#x5176;&#x4E2D;s&#x4EE3;&#x8868;&#x6539;&#x5B50;&#x6811;&#x7684;&#x6839;&#x8282;&#x70B9;&#xFF0C;l&#x4E3A;&#x5DE6;&#x5B50;&#x6811;&#x7684;&#x7F16;&#x53F7;&#xFF0C;r&#x4E3A;&#x53F3;&#x5B50;&#x6811;&#x7684;&#x7F16;&#x53F7;
//&#x7279;&#x522B;&#x8981;&#x6CE8;&#x610F;&#x5C0F;&#x4E8E;&#x53F7;<&#x7684;&#x91CD;&#x8F7D;&#xFF0C;&#x8FD9;&#x4E2A;&#x4F1A;&#x4E25;&#x91CD;&#x5F71;&#x54CD;map&#x7684;&#x5B58;&#x50A8;&#x65B9;&#x5F0F;&#xFF0C;&#x800C;&#x4E14;&#x8FD9;&#x4E2A;&#x91CD;&#x8F7D;&#x53EF;&#x4F5C;&#x4E3A;&#x4EE5;&#x540E;&#x7684;&#x6A21;&#x677F;&#xFF0C;&#x8BF7;&#x4ED4;&#x7EC6;&#x6B23;&#x8D4F;
struct Node{
    char *s;
    int l,r;
    Node(char *s,int l = 0,int r = 0):l(l),r(r){
        this->s = s;
    }
    bool operator < (Node const& a) const {
        if(strcmp(this->s,a.s) < 0) return true;
        if(strcmp(this->s,a.s) == 0 && this->l < a.l) return true;
        if(strcmp(this->s,a.s) == 0 && this->l == a.l) return this->r < a.r;
        return false;
    }
};

//mapn&#x7528;&#x4E8E;&#x6620;&#x5C04;&#xFF0C;lson&#x548C;rson&#x5B58;&#x50A8;&#x6811;&#xFF0C;_lson&#x548C;_rson&#x7528;&#x6765;&#x8BB0;&#x5F55;&#x6620;&#x5C04;&#x7F16;&#x53F7;&#x4E4B;&#x540E;&#x6240;&#x751F;&#x6210;&#x7684;&#x65B0;&#x6811;
map<Node, int> mapn;
int T,lson[maxn],rson[maxn],_lson[maxn],_rson[maxn],nod_,node_sum,ii;
char nod[maxn][5],str[500000];

//&#x8FD9;&#x4E2A;&#x662F;&#x6839;&#x636E;&#x8868;&#x8FBE;&#x5F0F;&#x7684;&#x5EFA;&#x6811;&#x8FC7;&#x7A0B;
int build_tree()
{
    int node_num = ++node_sum;
    sscanf(str+ii,"%[a-z]",nod[node_num]);
    lson[node_num] = -1;
    rson[node_num] = -1;
    ii += strlen(nod[node_num]);
    if(str[ii] == &apos;(&apos;){
        ++ii;
        lson[node_num] = build_tree();
        rson[node_num] = build_tree();
    }
    if(str[ii] == &apos;)&apos; || str[ii] == &apos;,&apos;) return node_num+(++ii)*0;
}

//&#x6839;&#x636E;&#x4E66;&#x4E2D;O(n)&#x7684;&#x7B97;&#x6CD5;dfs&#x4E00;&#x904D;&#xFF0C;&#x5E76;&#x4E3A;&#x6BCF;&#x4E2A;&#x5B50;&#x6811;&#x6620;&#x5C04;&#x7F16;&#x53F7;&#xFF0C;&#x5E76;&#x7528;_lson&#xFF0C;_rson&#x5B58;&#x4E0B;&#x6765;
//&#x6CE8;&#x610F;ii&#x5728;&#x8FD9;&#x4E2A;&#x51FD;&#x6570;&#x4E2D;&#x7684;&#x4F7F;&#x7528;&#xFF0C;&#x53EF;&#x4EE5;&#x81EA;&#x5DF1;&#x624B;&#x52A8;&#x6A21;&#x62DF;&#x4E00;&#x904D;&#xFF0C;&#x5B83;&#x80FD;&#x591F;&#x5B8C;&#x6210;&#x9898;&#x76EE;&#x6240;&#x8981;&#x6C42;&#x7684;&#x6B63;&#x786E;&#x7684;&#x7F16;&#x53F7;
int dfs(int root)
{
    Node _node(nod[root],-1,-1); int i = ii;
    _lson[root] = _rson[root] = -1;
    if(lson[root] != -1) _node.l = _lson[root] = dfs(lson[root]+(++ii)*0);
    if(rson[root] != -1) _node.r = _rson[root] = dfs(rson[root]+(++ii)*0);
    if(!mapn.count(_node)) mapn[_node] = i;
    else --ii;
    return mapn[_node];
}

//&#x8F93;&#x51FA;&#x51FD;&#x6570;&#xFF0C;&#x4E5F;&#x8981;&#x6CE8;&#x610F;&#x4E00;&#x4E0B;ii&#x7684;&#x4F7F;&#x7528;&#x65B9;&#x6CD5;&#xFF0C;&#x7528;&#x4E8E;&#x8BB0;&#x5F55;&#x5DF2;&#x7ECF;&#x8F93;&#x51FA;&#x7684;&#x5B57;&#x6BCD;&#x4E2A;&#x6570;
//&#x5982;&#x679C;&#x540E;&#x9762;&#x53D1;&#x73B0;&#x5B50;&#x6811;&#x7684;&#x6620;&#x5C04;&#x7F16;&#x53F7;&#x6BD4;ii&#x5C0F;&#xFF0C;&#x90A3;&#x4E48;&#x53EF;&#x4EE5;&#x77E5;&#x9053;&#x8FD9;&#x4E2A;&#x5B50;&#x6811;&#x51FA;&#x73B0;&#x8FC7;&#x4E86;&#xFF0C;&#x6240;&#x4EE5;&#x53EF;&#x4EE5;&#x7528;&#x7F16;&#x53F7;&#x4EE3;&#x66FF;&#xFF0C;&#x4EE5;&#x6D88;&#x9664;&#x8868;&#x8FBE;&#x5F0F;
void print(int root)
{
    
    printf("%s",nod[root]); ++ii;
    if(lson[root]!=-1&&rson[root]!=-1){
        printf("(");
        if(_lson[root] <= ii) printf("%d",_lson[root]);
        else print(lson[root]);
        printf(",");
        if(_rson[root] <= ii) printf("%d",_rson[root]);
        else print(rson[root]);
        printf(")");
    }
}

int main()
{
    scanf("%d",&T);
    while(T--)
    {
        scanf("%s",str);
        node_sum = 0; ii = 0;
        build_tree();
        mapn.clear(); ii = 1;
        dfs(1);
        ii = 0;
        print(1);
        printf("\n");
    }
     return 0;
}
