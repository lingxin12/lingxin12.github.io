A题：

题意：

给一个x，求满足gcd(a,b)+lcm(a,b)=x的a,b，答案保证总有解。

题解：

gcd(a,b)+lcm(a,b)=x,gcd(a,b)=d ==>

令a=da’ , b=db’ ( gcd(a’,b’) = 1 )

d+ab/d = x ==> d+da’b’ = x ==> x = d(1+a’b’)

只需要将x在sqrt(x)内分解成两个因子，然后求出a’*b’，然后在sqrt(ab)内分解两个因子使得gcd(a’,b’) = 1即可。

代码：

#include<bits/stdc++.h>
using namespace std;
int T,x;
int gcd(int a,int b){ return a%b==0?b:gcd(b,a%b); }
int main()
{
    scanf("%d",&T);
    while(T--)
    {
        scanf("%d",&x);
        int d = 1, qx = sqrt(x);
        int res_a,res_b,ok = 0;
        while(d <= qx){
            if(x%d == 0){
                int _ab = x/d - 1;
                int _a = 1 ,_qa = sqrt(_a);
                while(_a <= _qa){
                    if(_ab%_a == 0){
                        int _b = _ab/_a;
                        if(gcd(_a,_b) == 1){
                            res_a = _a,res_b = _b;
                            ok = 1;
                        }
                        if(ok) break;
                    }
                }
                if(ok) break;
            }
        }
        printf("%d %d\n",res_b,res_a);
    }
    return 0;
}

正解：

其实可以明显发现，a=1,b=x-1。

B题：

题意：

给定一个有n个元素的数组a，将a依次重复排列n次，如a={1,2,3}，排列后变成{1,2,3,1,2,3,1,2,3}，此时求新数组的最长严格上升子序列。

题解：

明显可以发现，问题可以转化成求数组a中最大的不重复元素子数组，简单说就是去重。

代码：

#include<bits/stdc++.h>
using namespace std;
map<int ,int > m;
int T,n,a,ans;
int main()
{
    scanf("%d",&T);
    while(T--)
    {
        scanf("%d",&n);
        m.clear();  ans = 0;
        for(int i=1;i<=n;i++){
            scanf("%d",&a);
            if(!m.count(a)) m[a] = 1;
            else ++ans;
        }
        printf("%d\n",n-ans);
    }
    return 0;
}

C题：

题意：

给一棵有n个节点的树，现在往树上的n-1条边上填数，只能填0到n-2，并且每个数只能用一次，同时要保证这棵树的所有MEX(u,v)的最大值尽量小。

（其中MEX(u,v)表示从u节点到v节点的路径上所有没有出现的数的最小值）

输入一个n，接下来n-1行，每行两个数u,v，表示u到v有一条边。

输出n-1行，对应输入的边，输出在这条边上填的数。

题解：

可以明显发现，当某个节点有三条边连着的时候，将0，1，2分别填入，则对于任意一条路径，0，1，2不会同时出现，满足题意。

#include<bits/stdc++.h>
using namespace std;
const int maxn = 1e5 + 3;
//vector<int > tree[maxn];
int n,x[maxn],y[maxn],nod[maxn];
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n-1;i++){
        scanf("%d%d",&x[i],&y[i]);
        //tree[x[i]].push_back(y[i]);  tree[y[i]].push_back(x[i]);
        ++nod[x[i]];  ++nod[y[i]];
    }
    int ok = 0,node_ok,node_ok_size;
    for(int i=1;i<=n;i++){
        if(nod[i] >= 3){
            ok = 1;  node_ok = i; node_ok_size = nod[i];
            break;
        }
    }
    if(ok == 0) for(int i=1;i<=n-1;i++) printf("%d\n",i-1);
    else{
        int ii = node_ok_size-1,jj=-1;
        for(int i=1;i<=n-1;i++){
            if(x[i] == node_ok || y[i] == node_ok) printf("%d\n",++jj);
            else printf("%d\n",++ii);
        }
    }
    return 0;
}
