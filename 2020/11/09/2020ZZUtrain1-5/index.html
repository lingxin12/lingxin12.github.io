A - Yet Another Palindrome Problem

（来源： codeforces - 1324B）

题意：

t个测试，对于每一个测试，给定一个n与一个长度为n的序列，求该序列存不存在一个长度至少为3的回文子序列？

题解：

很简单，只需要验证序列中是否存在两个相同的数字并且他们不是相邻的即可。

代码：

#include<bits/stdc++.h>
#define enl puts("")
#define pii pair<int,int>
#define pb push_back
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
const int maxn = 5002;

int t,n,book[maxn];
struct Node{
    int id,data;
}a[maxn];

bool cmp(struct Node a,struct Node b) { return a.data<b.data||(a.data==b.data&&a.id<b.id); }

int main()
{
    scanf("%d",&t);
    while(t--)
    {
        memset(book,0,sizeof(book));
        scanf("%d",&n);
        for(int i=1;i<=n;i++) {
            scanf("%d",&a[i].data);
            a[i].id = i;
            ++book[a[i].data];
        }
        sort(a+1,a+n+1,cmp);
        int ok = 0;
        for(int i=1;i<=n;) {
            if(book[a[i].data]>=2) {
                for(int j=i+1;j<=i+book[a[i].data]-1;j++) if(a[j].id!=a[i].id+1) {
                    ok = 1;
                    break;
                }
                if(ok) break;
            }
            i+=book[a[i].data];
        }
        printf("%s\n",ok?"YES":"NO");
    }
    return 0;
}

B - Forest Program

(来源：Gym - 102361F)

题意：

给定一个有n个结点，m条边的图，其中删去一些边，或者不删，可以变成森林（很多棵树）。

该图有某些特征，无重边，无自环，每一条边最多存在于图内的一个环中。

问有几种删法可以使图变成森林？（其中不同删法当且仅当存在两条不同的删除的边）

题解：

一道dfs题目，从某一个点标号为1，继续往下走，每走一步标号增加。不能往回走，也不能往标号小的点走，如果走的下一个点的标号比当前标号小，那么相减即为这个环的边数。

设有s个环，环长分别为，len1，len2······lens，孤立环的数量为len。

那么答案即为，(2^(len1)-1)*(2^(len2)-1)······(2^(lens)-1)*(2^len)。

另外，图不一定是连通的，所以要分别算出答案并相乘。

感悟：

这道题还有有一点点坎坷的。首先是题目漏看了一个条件：每一条边最多存在于图内的一个环中。导致自己一直在想如果两个环有相同边怎么办。然后是dfs的时候，会出现倒退或者是访问了一个比自己标号大，这里加了判断条件就行了（见代码注释）。另外int和long long的运算可能会出现问题，因为把数据全改为long long后就过了数据点4。最后卡在了数据点15，因为我没有意识到图可能不连通，最后加一个for循环每一个连通子图即可。

代码：

#include<bits/stdc++.h>
#define enl puts("")
#define pii pair<int,int>
#define pb push_back
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
const int maxn = 3e5+2;
const ll mod = 998244353;

ll n,m,u,v,book[maxn],cyclelen,mlen;
vector<ll > G[maxn];
ll res = 1;

ll fastpow(ll a,ll b)
{
    ll ans = 1;
    while(b) {
        if(b&1) ans = (ans*a)%mod;
        b>>=1;
        a = (a*a)%mod;
    }
    return ans;
}

void dfs(ll nod,ll dis,ll front)
{
    book[nod] = dis;
    for(int i=0;i<G[nod].size();i++) {
        //&#x8282;&#x70B9;&#x8BBF;&#x95EE;&#x4E0D;&#x80FD;&#x5012;&#x9000;&#x6216;&#x8005;&#x8BBF;&#x95EE;&#x5230;&#x4E00;&#x4E2A;&#x6BD4;&#x81EA;&#x5DF1;&#x6807;&#x53F7;&#x5927;&#x7684;&#x8282;&#x70B9;&#xFF0C;&#x9632;&#x6B62;&#x91CD;&#x590D;
        if(book[G[nod][i]]>dis||G[nod][i]==front) continue;
        ++mlen;
        if(book[G[nod][i]]>0) {
            cyclelen += dis-book[G[nod][i]]+1;
            res = (res*(fastpow( 2, dis-book[G[nod][i]]+1 )-1))%mod;
        }
        else if(book[G[nod][i]]==0) dfs(G[nod][i],dis+1,nod);
    }
}

int main()
{
    scanf("%lld%lld",&n,&m);
    for(int i=1;i<=m;i++) {
        scanf("%lld%lld",&u,&v);
        G[u].pb(v);
        G[v].pb(u);
    }
    for(int i=1;i<=n;i++) if(book[i] == 0) {
        cyclelen = mlen = 0;
        dfs(i,1,-1);
        mlen -= cyclelen;
        res = (res*fastpow(2,mlen))%mod;
    }
    printf("%lld\n",res);
    return 0;
}

C - Invoker

(来源：HDU - 6739)

题意：

大概讲的是dota2里面有这么一个英雄，QWE三个技能不同的组合使得R技能所产生的效果不相同。

然后给出一个长度最多为1e5的R技能字符串s，然后求出打出这套技能所需要最少的摁键数。

比如说“CX”这两个技能，最少的摁键数为6，为WWWRQR。

题解：

本题是一个简单的dp。

设dp[i][j]为施放到第i个技能的第j个排列所需要的最小长度。

很明显可以找到转移方程：

dp[i][j] = min{dp[i][j],dp[i-1][k]+diff}

其中diff代表在第i-1个技能的第k个排列，要施放第i-1个技能的第j个排列好需要点下几个摁键。

这是一种看上去比较暴力的dp，因为他需要考虑所有的排列，但是与普通的暴力（普通的暴力就是要考虑所有技能的排列）不同的是，每一次转移只需要考虑前一个与后一个的排列，每次转移最多进行6x6次比较，所以其实复杂度为O(n)。

感悟：

刚开始有想过用dp，但是对dp不熟，所以一直不知道怎么转移，而且有一点魔怔了，总想着绝对要在n次内出答案，其实对于1e5的数据，可以去尝试暴力一点，不要总想着有什么技巧可以快速确定下一个该如何排列为最佳。后面思考了一会，总得来说，各个知识点的练习还是不够吧。

B - Nauuo and Circle

（来源： codeforces - 1173D）

题意：

给定有n个节点的一棵树。将树的每一个节点放在一个圆上，使得所有边不相交，一共有几种放法？

比如有4个节点 ，他们的边分别为1-2，1-3，2-4，那么有16种放法。

题解：

标准答案是树形dp，但是我并没有看懂，我觉得这个问题只是一个存粹的数学问题。可以发现答案为所有节点的度数阶乘的乘积乘上节点数。下面用手简单证明了一下。

代码：

#include<bits/stdc++.h>
#define enl puts("")
#define pii pair<int,int>
#define pb push_back
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
const int maxn = 2e5 + 2;
const ll mod = 998244353;

int n,a,b,son[maxn];
ll A[maxn],res;

int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n-1;i++) {
        scanf("%d%d",&a,&b);
        ++son[a];
        ++son[b];
    }
    A[0] = A[1] = 1;
    for(int i=2;i<=n;i++) A[i] = A[i-1]*i%mod;
    res = 1;
    for(int i=1;i<=n;i++) res = res*A[son[i]]%mod;
    printf("%lld\n",n*res%mod);
    return 0;
}

A - Dubious Document 2

（来源： AtCoder - abc076_c）

题意：

给一个只有小写字母组成的字符串S，将其中的一些字母换成’?’，生成了一个由小写字母和’?’组成的字符串S’。现在给一个字符串T。现在根据下面两种情况由S’生成S（即将问号换成小写字母）。

1.T为S的子串。

2.在满足条件1的情况下，使S的字典序最小。

输出S即可。（|S| ≤ 50）

例如。

S’=?tc????，T=coder，那么S=atcoder。

S’=??p??d??，T=abc，那么不存在S，输出UNRESTORABLE。

题解：

普通的字符串匹配问题，暴力解决，把’?’当成任意字符匹配，找到最后一个能放T的位置，然后剩余的’?’填上’a’即可。

代码：

#include
#define enl puts("")
#define pii pair
#define pb push_back
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
char s[100],t[100];
int main()
{
    scanf("%s%s",s+1,t+1);
    int len1 = strlen(s+1), len2 = strlen(t+1);
    int i = len1-len2+1;
    int okok = 0;
    while(i>=1)
    {
        int ok = 1,fla = 0,chai;
        for(int k=i,j=1;j<=len2;j++,k++) {
            if(s[k]!='?' && s[k]!=t[j]) {
                ok = 0;
                break;
            }
            if(j == len2 && fla == 0) {
                fla = 1;
                chai = i;
            }
        }
        if(ok)
        {
            for(int k=1;k<=chai-1;k++) 
                if(s[k] == '?') printf("a"); 
                else printf("%c",s[k]); 
            for(int k=chai;k<=chai+len2-1;k++) printf("%c",t[k-chai+1]);
            for(int k=chai+len2;k<=len1;k++) 
                if(s[k] == '?') printf("a"); 
                else printf("%c",s[k]);
            enl;
            okok = 1;
            break;
        }
        --i;
    }
    if(!okok) printf("UNRESTORABLE\n");
    return 0;
}
