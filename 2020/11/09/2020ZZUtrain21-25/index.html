A - Key races

题意：

打字比赛，给出两个人的延迟和打字速度。

题解：

比较即可。

代码：

#include<bits/stdc++.h>
#define enl puts("")
#define pii pair<int,int>
#define pb push_back
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;

int main()
{
    scanf("%d%d%d%d%d",&s,&v1,&v2,&t1,&t2);
    int res1 = v1*s+2*t1;

B - The number on the board

题意：

刚开始有一个数字，其数位和不少于k，现要改变某些数位，使其变成数n，求最少改变的数位个数。

1<=k<=10^9，1<=n<=10^100000。

题解：

如果n各个数位之和不少于k，答案即为0。若不是，各个数位从小到大排序，依次变成9，满足条件跳出循环。

代码：

#include<bits/stdc++.h>
#define enl puts("")
#define pii pair<int,int>
#define pb push_back
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
const int maxn = 1e5+2;
int k,len;
char s[maxn];
int sn[maxn];
int main()
{
    scanf("%d",&k);
    scanf("%s",s+1);
    len = strlen(s+1);
    for(int i=1;i<=len;i++) sn[i] = s[i]-&apos;0&apos;;
    sort(sn+1,sn+len+1);
    int tot = 0;
    for(int i=1;i<=len;i++) tot+=sn[i];
    if(tot >= k) {
        printf("0\n");
        return 0;
    }
    int res = 0;
    for(int i=1;i<=len;i++) {
        tot += 9-sn[i];
        ++res;
        if(tot >= k) break;
    }
    printf("%d\n",res);
    return 0;
}

C - SA-IS后缀数组

题意：

1<=i<=n-1，对一个字符串，比较s[i…n]与s[i+1…n]的字典序大小。（水题）

题解：

从后往前比较即可。

代码：

#include<bits/stdc++.h>
#define enl puts("")
#define pii pair<int,int>
#define pb push_back
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
const int inf = 0x7fffffff;
const int maxn = 1e6+2;
int T,n;
char s[maxn],res[maxn];

int main()
{
    scanf("%d",&T);
    while(T--)
    {
        scanf("%d%s",&n,s+1);
        int i = n-1, j = n, fu = s[i]<s[j]?0:1;
        res[i] = fu?&apos;>&apos;:&apos;<&apos;;
        --i,--j;
        for(;i>=1;--i,--j)
        {
            if(s[i]!=s[j]) fu = s[i]<s[j]?0:1;
            res[i] = fu?&apos;>&apos;:&apos;<&apos;;	
        }
        res[n] = &apos;\0&apos;;
        printf("%s\n",res+1);
    }
    return 0;
}
