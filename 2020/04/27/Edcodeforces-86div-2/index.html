A题：

题意：

给定大于等于0的正整数x,y和大于等于1的正整数a,b，花费a元使x或y增加（减少）1，也可以花费b元使x和y同时增加（减少）1，求使x，y变为0的最少花费。

题解：

假设x>=y，有两种情况，将x降至y，再一起降至0，或者分别降至0。

同时易证将某数升至某值再降等情况均不如上述两种情况中某一种要好。

代码：

include<bits/stdc++.h>
#define enl puts("")
#define lop(i,a,b) for(int i=(a);i<(b);i++)
#define pol(i,a,b) for(int i=((b)-(1));i>=(a);i--)
#define pii pair<int,int>
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;

ll x,y,a,b,t;

int main()
{
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld%lld%lld%lld",&x,&y,&a,&b);
        if(x < y) swap(x,y);
        ll c = y*b+(x-y)*a, d = (x+y)*a;
        if(c > d) printf("%lld\n",d);
        else printf("%lld\n",c);
    }
    return 0;
}

C题：

题意：

给定a,b,q，其中包含q个询问，每次询问都有l，r，求在[l,r]内有几个数满足

((x mod a) mod b) ≠((x mod b) mod a)

题解：

通过列举我们可以发现，（假设a>b，若a=b为答案为0）若令c=lcm(a,b)，则[k*c,k*c+a-1]内的数都不满足上式，其中k=0,1,2,3……

可以特判一下a=b，a mod b = 0的情况，答案都为0。

问题转化成了区间[l,r]内去掉区间[k*c,k*c+a-1]内的数后还剩下几个数。

代码：

#include<bits/stdc++.h>
#define enl puts("")
#define lop(i,a,b) for(int i=(a);i<(b);i++)
#define pol(i,a,b) for(int i=((b)-(1));i>=(a);i--)
#define pii pair<int,int>
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
ll a,b,T,l,r,q;
ll gcd(ll a,ll b){ return a%b?gcd(b,a%b):b; }
ll lcm(ll a,ll b){ return a/gcd(a,b)*b; }
int main()
{
    //freopen("data.out","w",stdout);
    scanf("%lld",&T);
    while(T--)
    {
        scanf("%lld%lld%lld",&a,&b,&q);
        if(a < b) swap(a,b);
        lop(qq,0,q)
        {
            if(qq!=0)printf(" ");
            scanf("%lld%lld",&l,&r);
            ll _lcm = lcm(a,b);
            //if(a == b) printf("0");
            ///else if(r < a) printf("0");
            //else if(r == a) printf("%d",(a%b!=0));
            //else if(r < _lcm) printf("%d",(a%b!=0));
            //else{
            if(a%b == 0){
                printf("0");
                continue;
            }
            ll sam;
            ll ln = l/_lcm, rn = r/_lcm;
            if(ln == rn){
                ll bas = ln*_lcm;
                sam = 0;
                if(r < bas+a) sam+=(r-l+1);
                //////////////&#x6F0F;&#x4E86;&#x4E00;&#x4E2A;&#x6761;&#x4EF6;&#xFF1A;l>bas+a&#xFF0C;&#x4E0D;&#x7136;&#x4F1A;&#x53D8;&#x6210;&#x8D1F;&#x6570;///////////
                else if(r >= bas+a && l<bas+a) sam+=(bas+a-l);
            }
            else{
                sam = (rn-ln-1)*a;
                if(l<ln*_lcm+a) sam+=(ln*_lcm+a-l);
                if(r<rn*_lcm+a) sam+=(r-rn*_lcm+1); else sam+=a;
            }
            printf("%lld",(r-l+1)-sam);
        }
        enl;
    }
    return 0;
}

D题：

题意：

给定正整数n,k(<=2e5)，再给一个长度为n的正整数序列m，该序列中mi<=k，最后给一个长度为k的序列c，并且保证n>=c1>=c2>=…..>=ck>=1，将m序列中的n个数分成ans组，使得每一组所有>=i的数的个数不超过ci。

求最小的ans值，并在每一行输出每一组元素个数和元素。

题解：

最开始的想法是这样的，将序列m排序后，从大到小循环，然后循环每一组，对于第j组来说，如果v[j].size()==c[m[i]]则该组已满（有这个条件后前面不会出现>的情况的）。继续循环下一组。

但是当ans值很大的时候，可能会出现复杂度高达O(n^2)的程度，所以需要改进。

注意到，如果此次c[m[i]]与上次c[m[i+1]]相同，那么维护一个now值表示第now组可以放，这样就不用每次都从第一次开始循环判断了。

代码：

#include<bits/stdc++.h>
#define enl puts("")
#define pb push_back
#define lop(i,a,b) for(int i=(a);i<(b);i++)
#define pol(i,a,b) for(int i=((b)-(1));i>=(a);i--)
#define pii pair<int,int>
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
const int maxn = 2e5 + 3;
int n,k,m[maxn],c[maxn],d[maxn];
vector<int > v[maxn];
int main()
{
    scanf("%d%d",&n,&k);
    lop(i,1,n+1) scanf("%d",&m[i]);
    lop(i,1,k+1) scanf("%d",&c[i]);
    sort(m+1,m+n+1);
    m[n+1] = 0;
    ///&#x6838;&#x5FC3;&#x4EE3;&#x7801;
    //front&#x8BB0;&#x5F55;&#x7684;&#x662F;c[m[i+1]],now&#x662F;&#x6B64;&#x65F6;&#x53EF;&#x4EE5;&#x653E;&#x7684;&#x90A3;&#x4E00;&#x7EC4;
    v[1].pb(m[n]); int front = c[m[n]], now = c[m[n]]==1?2:1;
    pol(i,1,n)
    {
        if(c[m[i]] != front) now = 1;
        v[now].pb(m[i]);
        front = c[m[i]];
        if(v[now].size() == front) now++;
    }
    
    
    int ans = 1;
    while(v[ans+1].size() != 0) ans++;
    printf("%d\n",ans);
    lop(i,1,ans+1){
        printf("%d",v[i].size());
        lop(j,0,v[i].size()) printf(" %d",v[i][j]);
        enl;
    }
    return 0;
}
