紫书例题6-20理想路径（Ideal Path UVA1599）题解

2020-12-18- 2021-03-07
题解 > 紫书-题解算法紫书

问题：

给定一个n个点m条边的无向图，每条边上都涂有1种颜色。求点1到点n的一条路径，使得经过的边数最少，在此前提下，经过边的颜色序列最小。可能有自环与重边。输入保证至少存在一条连接1和n的道路。

2≤n≤10e5,1≤m≤2×10e5,1≤ci≤10e9。

（本题有多组输入）

分析：

这道题重点在于在求出最短路之后如何再次找到最短路从而筛选出最小字典序。

通常我们使用的方式都是记录父节点的方式来寻找最短路径。但是实际上本题由于每条边的长度都为1，所以可以通过最短路径上下一个节点的d值一定会与本节点相差1。这是一种比较好的寻找方法。

（这里的d值指的是单源最短路径中的d数组）

所以可以通过一次bfs，求出每一个节点的d值。再用一次bfs来找最小字典序。具体找法如下：

对于队列q中的节点，先遍历一遍，找到所有属于最短路径中的边中的最小的颜色mincolor，然后再遍历一遍，把队列q中的节点的所有满足最短路径条件（d[to[i]]==d[from]-1）的所有颜色等于mincolor的边所指向的节点都入队，然后重复操作。

这里有两个比较需要注意的优化：

1.遍历队列q的方法：先求出q.size()，再通过访问q.front()后q.push(q.front()),q.pop()来遍历。

2.这一点非常非常重要，同时它也是卡了我一天的重要原因。就是一定要注意避免同一个节点多次入队。在这个无向图中，第一次bfs出来的最短路径可能有很多条，如果有很多条最短路径都有同一个交点，那么这个交点就会多次入队，从而重复计算，应该用vis来标记哪些节点入过队。

（真是恶心人的数据）

感想：

这道题搞了两天，这我是没有想到的。主要是上面第2点一直没有意识到。导致我花了大量的时间一直在进行托马斯回旋大优化，甚至还一度认为是STL存图在卡我时间，第二天我又重写的代码，用C语言来写的的存图（用的是模拟邻接链表），不过还是没有AC，看了别人的代码之后，看到了vis数组，瞬间意识到了上面第二点的那个问题，原来正解和语言真的没有关系（哭 ！）

（不过用C语言是真的快，这道题给了5秒时间，STL存图几百ms，C语言存图80ms）

代码：

#include<bits/stdc++.h>
#define enl puts("")
#define pii pair<int,int>
#define pb push_back
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
const int maxn = 1e5 + 10;
const int maxm = 4e5 + 10;
const int inf = 0x3f3f3f3f;

int d[maxn],fir[maxn], color[maxm] , to[maxm] , nxt[maxm], resk, rescolor[maxn],vis[maxn];
int n,m,a,b,c,edgei;
queue<int > q;

inline void intit()
{
    memset(fir,-1,sizeof(fir));
    memset(vis,0,sizeof(vis));
    edgei = 0;
    while(!q.empty()) q.pop();
}

inline void addedge(int a,int b,int c)
{
    nxt[++edgei] = fir[a];
    to[edgei] = b; color[edgei] = c;
    fir[a] = edgei;
}

inline void solve()
{
    while(!q.empty())
    {
        int from = q.front(); q.pop();
        for(int i=fir[from];i!=-1;i=nxt[i]) {
            if(d[to[i]] > d[from]+1) {
                d[to[i]] = d[from]+1;
                q.push(to[i]);
            }
        }
    }
    resk = d[1];
}

inline void bfs()
{
    for(int kk=1;kk<=resk;kk++)
    {
        int qlen = q.size(), mincolor = inf;
        for(int k=1;k<=qlen;k++) {
            int from = q.front(); q.push(q.front()); q.pop();
            for(int i=fir[from];i!=-1;i=nxt[i]) if(d[to[i]]==d[from]-1) mincolor = min(mincolor, color[i]);
        }
        rescolor[kk] = mincolor;
        if(kk == resk) return;
        for(int k=1;k<=qlen;k++) {
            int from = q.front(); q.pop();
            for(int i=fir[from];i!=-1;i=nxt[i]) 
                if(color[i] == mincolor && d[to[i]]==d[from]-1 && !vis[to[i]])
                    q.push(to[i]), vis[to[i]] = 1;
        }
    }
}

int main()
{
    while(~scanf("%d%d",&n,&m))
    {
        intit();
        for(int i=1;i<=m;i++) {
            scanf("%d%d%d",&a,&b,&c);
            if(a == b) continue;
            addedge(a,b,c);
            addedge(b,a,c);
        }
        for(int i=1;i<=n;i++) d[i] = inf;
        d[n] = 0;
        q.push(n);
        solve();
        q.push(1);
        bfs();
        printf("%d\n",resk);
        for(int i=1;i<=resk;i++) printf("%d%c",rescolor[i],i==resk?&apos;\n&apos;:&apos; &apos;);
    }
    return 0;
}
