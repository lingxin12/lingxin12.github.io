A AquaMoon and Two Arrays

题意：

给定两个长度为n的数组a，b，每一次操作可以在a数组中选择任意两个位置i，j，然后a[i]-1，a[j]+1，问多少次操作之后能够把a数组变成b数组，并输出方案（每行输出i和j）。不要求操作次数最少，但是要求要在100次操作以内。

1 <= t <= 100, 1 <= n <= 100, 0 <= ai <= 100, 0 <=bi <= 100

题解：

遍历a数组，把a[i]<b[i]与a[i]>b[i]的位置先记下来，每次进行相对应的操作即可

代码：

#include<bits/stdc++.h>
#define enl cout << "\n"
#define jud cout << "caonima" << endl;
#define pii pair<int,int>
#define pll pair<long long,long long>
#define pdd pair<double , double >
#define pss pair<string , string >
#define pb push_back
#define x first
#define y second
#define Clear(x) while(!x.empty())x.pop()
#define pqg(type) priority_queue<type,vector<type>,greater<type>>
#define pql(type) priority_queue<type,vector<type>,less<type>>
#define Max(a,b) ((a)<(b)?(b):(a))
#define Min(a,b) ((a)<(b)?(a):(b))
int T;
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
const int maxn = 1e5 + 3;

int n, a[110], b[110];
vector<int > vmn, vmx;
vector<pii > res;

void solve()
{
    vmn.clear();
    vmx.clear();
    res.clear();
    scanf("%d", &n);
    for(int i = 1; i <= n; i ++) scanf("%d", a + i);
    for(int i = 1; i <= n; i ++) scanf("%d", b + i);

    for(int i = 1; i <= n; i ++)
        if(a[i] > b[i]) vmx.pb(i);
        else if(a[i] < b[i]) vmn.pb(i);

    int i, j;
    for(i = 0, j = 0; i < vmn.size() && j < vmx.size(); )
    {
        int x = vmn[i], y = vmx[j];
        while(1) {
            res.pb({y, x});
            -- a[y]; ++ a[x];
            if(a[y] == b[y]) {
                ++ j;
                if(a[x] == b[x]) ++ i;
                break;
            }
            if(a[x] == b[x]) {
                ++ i;
                break;
            }
        }
    }

    if(i == vmn.size() && j == vmx.size()) {
        printf("%d\n", res.size());
        for(auto v : res) printf("%d %d\n", v.x, v.y);
    }
    else puts("-1");
}

int main()
{
    //freopen("data.in","r",stdin);
    //ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    scanf("%d", &T);
    //T = 1;
    while(T--)
        solve();
    return 0;
}

B AquaMoon and Stolen String

题意：

给定n个长度为m的字符串，n为奇数，把n个字符串两两配对，配成(n - 1) / 2对之后多出一个字符串去掉，然后对于每一对字符串，可能会在对应的位置交换字符（比如aaaa与bbbb，可能会被换成abab与baba），再给你n个已经配对好并且换好的n个字符串，输出被去掉的字符串是哪一个。

1 <= t <= 100, 1 <= n <= 1e5, 1 <= m <= 1e5, all n <= 1e5

题解：

题目看起来复杂，实际上只需要记录第i个字符上26个小写字母的数量，比对去掉那个字符串前后看看是哪个小写字母多出来了，那么去掉的那个字符串第i个字符就是多出来的那一个字母。

代码：

#include<bits/stdc++.h>
#define enl cout << "\n"
#define jud cout << "caonima" << endl;
#define pii pair<int,int>
#define pll pair<long long,long long>
#define pdd pair<double , double >
#define pss pair<string , string >
#define pb push_back
#define x first
#define y second
#define Clear(x) while(!x.empty())x.pop()
#define pqg(type) priority_queue<type,vector<type>,greater<type>>
#define pql(type) priority_queue<type,vector<type>,less<type>>
#define Max(a,b) ((a)<(b)?(b):(a))
#define Min(a,b) ((a)<(b)?(a):(b))
int T;
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
const int maxn = 1e5 + 10;

int n, m, cnt;
char ins[maxn], res[maxn];
int od[maxn][26], nw[maxn][26];

void solve()
{
    memset(od, 0, sizeof od);
    memset(nw, 0, sizeof nw);
    cnt = -1;
    scanf("%d%d", &n, &m);

    for(int i = 1; i <= n; i ++) {
        scanf("%s", ins);
        for(int j = 0; j < m; j ++) ++ od[j][ins[j] - &apos;a&apos;];
    }

    for(int i = 1; i <= n - 1; i ++) {
        scanf("%s", ins);
        for(int j = 0; j < m; j ++) ++ nw[j][ins[j] - &apos;a&apos;];
    }

    for(int i = 0; i < m; i ++) {
        for(int j = 0; j < 26; j ++)
            if(od[i][j] == nw[i][j] + 1)
                res[++ cnt] = j + &apos;a&apos;;
    }

    res[++ cnt] = &apos;\0&apos;;
    printf("%s\n", res);
}

int main()
{
    scanf("%d", &T);
    //T = 1;
    while(T--)
        solve();
    return 0;
}

C AquaMoon and Strange Sort

题意：

给定一个长度为n的序列，序列上的每一个数刚开始都是朝向右边，每一次操作可以选择两个相邻的数并交换它们，同时改变这两个数的朝向（左边变向右边，右边变向左边），问是否能够通过一系列操作使该序列变成非严格上升序列，同时最后每一个数的都朝向右边。

1 <= t <= 50, 1 <= n <= 1e5, 1 <= ai <= 1e5, all n <= 1e5

题解：

对于每一个数，它不管换到哪里，只有换到与原来位置奇偶性相同的位置朝向才能向右，那么将所有数排序之后，把相同的数当成一块，然后与原来的位置比较，奇数和偶数个数不同则不能操作成功。

代码：

#include<bits/stdc++.h>
#define enl cout << "\n"
#define jud cout << "caonima" << endl;
#define pii pair<int,int>
#define pll pair<long long,long long>
#define pdd pair<double , double >
#define pss pair<string , string >
#define pb push_back
#define x first
#define y second
#define Clear(x) while(!x.empty())x.pop()
#define pqg(type) priority_queue<type,vector<type>,greater<type>>
#define pql(type) priority_queue<type,vector<type>,less<type>>
#define Max(a,b) ((a)<(b)?(b):(a))
#define Min(a,b) ((a)<(b)?(a):(b))
int T;
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
const int maxn = 1e5 + 3;

int n, a[maxn];
vector<pii > ve;

void solve()
{
    ve.clear();
    ve.pb({0, 0});

    scanf("%d", &n);
    for(int i = 1; i <= n; i ++) scanf("%d", a + i), ve.pb({a[i], i});
    sort(a + 1, a + n + 1);
    sort(ve.begin(), ve.end());

    int cnto  = 1, cnte = 0;
    for(int i = 2; i <= n + 1; i ++)
    {
        if(i == n + 1 || a[i] != a[i - 1])
        {
            int odd = 0, even = 0;
            for(int j = i - 1; j >= i - cnto - cnte; j --)
                if(ve[j].y & 1) ++ odd;
                else ++ even;
            
            if(even != cnte || odd != cnto) {
                puts("NO");
                return ;
            }

            if(i & 1) cnto = 1, cnte = 0;
            else cnte = 1, cnto = 0;
        }
        else if(i & 1) ++ cnto;
        else ++ cnte;
    }

    puts("YES");
}

int main()
{
    //freopen("data.in","r",stdin);
    //ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    scanf("%d", &T);
    //T = 1;
    while(T--)
        solve();
    return 0;
}

D AquaMoon and Chess

题意：

有一个长度为n的一维棋盘，棋盘上有些地方有棋子，有些地方没有，同一个位置最多有有一个棋子，类似跳棋规则，如果一个棋子向左跳，那么它左边必然要有一个棋子，然后它跳到它左边棋子的左边，右边同理。求通过这种规则最后形成的棋盘有几种？（mod 998244353）

1 <= t <= 1e4, 1 <= n <= 1e5, all n <= 1e5

题解：

把两个连着的棋子合在一起（比如有3个，那么合2个留下一个），设最后两个连着的棋子有sum2个，没有棋子的地方有sum0个，答案为C(sum0 + sum2, sum0)。

若样例：010 答案为1

0110 转化为 020 答案为C(3, 2)

01110 转化为 0210 仔细考察所有方案可以发现实际上1并没有用，一样还是020排列组合。

其余情况类似

因为mod为质数，先预处理n!和inv(n!)然后用逆元求组合数。

代码：

#include<bits/stdc++.h>
#define enl cout << "\n"
#define jud cout << "caonima" << endl;
#define pii pair<int,int>
#define pll pair<long long,long long>
#define pdd pair<double , double >
#define pss pair<string , string >
#define pb push_back
#define x first
#define y second
#define Clear(x) while(!x.empty())x.pop()
#define pqg(type) priority_queue<type,vector<type>,greater<type>>
#define pql(type) priority_queue<type,vector<type>,less<type>>
#define Max(a,b) ((a)<(b)?(b):(a))
#define Min(a,b) ((a)<(b)?(a):(b))
int T;
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
const int maxn = 1e5 + 10, p = 998244353;

int n, sum0, sum2, a[maxn], book[maxn], ainv[maxn], s[maxn];

int fpow(int a, int b)
{
    ll res = 1;
    while(b) {
        if(b & 1) res = res * a % p;
        b >>= 1;
        a = 1ll * a * a % p;
    }
    return res;
}


void init()
{
    ainv[0] = ainv[1] = 1;
    for(int i = 2; i <= 1e5; i ++)
        ainv[i] = 1ll * ainv[i - 1] * fpow(i, p - 2) % p;
    s[0] = s[1] = 1;
    for(int i = 2; i <= 1e5; i ++)
        s[i] = 1ll * s[i - 1] * i % p;
}

inline int C(int a, int b)
{
    return 1ll * s[a] * ainv[b] % p * ainv[a - b] % p;
}

void solve()
{
    sum0 = sum2 = 0;
    memset(book, 0, sizeof book);
    scanf("%d", &n);
    for(int i = 1; i <= n; i ++) scanf("%1d", a + i);
    for(int i = 1; i <= n; i ++) {
        if(a[i] == 0) ++ sum0;
        if(a[i] == 1 && a[i - 1] == 1 && book[i] == 0 && book[i - 1] == 0) {
            book[i - 1] = book[i] = 1;
            ++ sum2;
        }
    }

    printf("%d\n", C(sum2 + sum0, sum0));
}

int main()
{
    init();
    //freopen("data.in","r",stdin);
    //ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    scanf("%d", &T);
    //T = 1;
    while(T--)
        solve();
    return 0;
}
