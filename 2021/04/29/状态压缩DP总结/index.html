前言

在DP中有一个比较重要的概念：状态。

在一个问题中，状态是多种多样的，表示状态的方法也有很多种。对于一个复杂或者比较占空间的状态，是否有一种方法能够简单地将其表示，显得非常重要。

比如一排有10000盏灯，有的亮有的暗，那么如何表示它们的状态。此时可以用book数组，’1’表示亮，’0’表示暗。如果有10000000000000盏灯那么数组就不够用了，此时可以用一个二进制数来表示这一排灯的状态，第i位为1表示亮，这样只需要一个整数就可以表示这一排灯的状态了。

除此之外，在状态进行转移的过程中，位运算有可能会给我们带来很多便捷之处。

常见位运算总结

去掉最后一位          | (101101->10110)           | x shr 1
在最后加一个0         | (101101->1011010)         | x shl 1
在最后加一个1         | (101101->1011011)         | x shl 1+1
把最后一位变成1       | (101100->101101)          | x or 1
把最后一位变成0       | (101101->101100)          | x or 1-1
最后一位取反          | (101101->101100)          | x xor 1
把右数第k位变成1      | (101001->101101,k=3)      | x or (1 shl (k-1))
把右数第k位变成0      | (101101->101001,k=3)      | x and not(1 shl (k-1))
右数第k位取反         | (101001->101101,k=3)      | x xor (1 shl (k-1))
取末三位             | (1101101->101)            | x and 7
取末k位              | (1101101->1101,k=5)       | x and (1 shl k-1)
取右数第k位           | (1101101->1,k=4)          | x shr (k-1) and 1
把末k位变成1          | (101001->101111,k=4)      | x or (1 shl k-1)
末k位取反            | (101001->100110,k=4)      | x xor (1 shl k-1)
把右边连续的1变成0     | (100101111->100100000)    | x and (x+1)
把右起第一个0变成1     | (100101111->100111111)    | x or (x+1)
把右边连续的0变成1     | (11011000->11011111)      | x or (x-1)
取右边连续的1         | (100101111->1111)         | (x xor (x+1)) shr 1

例题

互不侵犯

(题目来源：https://www.luogu.org/problemnew/show/P1896)

题意

在n行n列的棋盘放k个国王，使得任意一个国王周围8个格子都没有其他国王，求方案数。

1<=n<=9，0<=k<=n*n

题解

国王个数是一个限制条件，显然需要把它当成一个状态。

把一行作为一个状态，按行进行递推（状压DP中常以行递推）。

对于每一行，国王的放法（不放也是一种方法），作为一种状态。

此时可以定义dp[i][j][s]为到第i行，第i行为第j个状态，前i行放的国王数量为s时的方案数。

那么答案就是Σdp[n][j][k]。

（如果你非常纠结这个状态方程是如何来的，那么是没有必要的，这道题是一个很简单的模板题，你只需要感觉这么设状态方程是没有错的就可以了，至于为什么不需要深究，随着你以后的深入学习，回头看这道题的题解时，自然而然会有更深的理解）

转移也非常简单：对于第i行第j个状态放s个国王，

有dp[i][j][s] = Σdp[i-1][t][s-sum1[j]]。

右边代表对于第i-1行所有的状态t，sum1[j]代表第j个状态放的国王数（其实就是第j个状态二进制中有几个1）。s-sum1[j]显然就是i-1行第t个状态放了s-sum1[j]个国王，如果第t个状态中1的个数不满足s-sum1[j]个国王，那么其实其dp值为0，此时不产生贡献，所以不需要多加一个判断。

那么在进行转移时，上一行的t状态和这一行的j状态肯定得要满足题目条件才行呀。

比如下面这种情况就是不满足的：

j状态：1010001
t状态：0100000

那么就得有一种方法判断t状态和j状态是否满足题意。

根据题目意思，其实就是下面的条件：

if(!(sta[j]&sta[t]) && !(sta[j]&(sta[t]<<1)) && !(sta[j]&(sta[t]>>1)))
//sta[i]&#x8868;&#x793A;&#x7B2C;i&#x4E2A;&#x72B6;&#x6001;&#x7684;&#x503C;&#xFF08;&#x5373;&#x4E0A;&#x8FF0;&#x6240;&#x8BF4;&#x7684;&#x538B;&#x7F29;&#x540E;&#x7684;&#x4E8C;&#x8FDB;&#x5236;&#x6570;&#xFF09;

如果你对上面的位运算感到困惑，那么请动一动手创造几组二进制数，进行运算看看结果。

那么接下来看具体代码如何实现的。

代码

#include<bits/stdc++.h>
#define enl cout << "\n"
#define pii pair<ll,int>
#define pdd pair<double , double >
#define pss pair<string , string >
#define pb push_back
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
int n,k,cnt = 0;
int sta[1500], sum1[1500];  //sta[i]&#x8868;&#x793A;&#x7B2C;i&#x4E2A;&#x72B6;&#x6001;&#x5BF9;&#x5E94;&#x7684;&#x4E8C;&#x8FDB;&#x5236;&#x6570;&#xFF0C;sum1[i]&#x8868;&#x793A;&#x7B2C;i&#x4E2A;&#x4E8C;&#x8FDB;&#x5236;&#x6570;&#x6709;&#x51E0;&#x4E2A;1
ll dp[11][1500][100];

void intit()
{
    for(int i=0;i<(1<<n);i++) {
        if(i&(i<<1)) continue;  //&#x6839;&#x636E;&#x9898;&#x610F;&#xFF0C;&#x540C;&#x4E00;&#x884C;&#x7684;&#x56FD;&#x738B;&#x4E0D;&#x80FD;&#x76F8;&#x90BB;&#xFF0C;&#x5982;&#x679C;&#x4E0D;&#x7406;&#x89E3;&#x8FD9;&#x4E2A;&#x4F4D;&#x8FD0;&#x7B97;&#xFF0C;&#x8D76;&#x7D27;&#x52A8;&#x52A8;&#x624B;&#xFF0C;&#x8FD9;&#x4E00;&#x6B65;&#x53EF;&#x4EE5;&#x53BB;&#x6389;&#x4E0D;&#x6EE1;&#x8DB3;&#x9898;&#x610F;&#x7684;&#x56FD;&#x738B;&#x653E;&#x7F6E;&#x72B6;&#x6001;
        int sum = 0;
        for(int j=0;j<n;j++) if(i&(1<<j)) ++sum;  //&#x8BA1;&#x7B97;&#x4E8C;&#x8FDB;&#x5236;&#x6570;&#x4E2D;&#x6709;&#x51E0;&#x4E2A;1
        sta[++cnt] = i;  //&#x7B2C;cnt&#x4E2A;&#x72B6;&#x6001;&#x7684;&#x4E8C;&#x8FDB;&#x5236;&#x6570;&#x4E3A;i
        sum1[cnt] = sum;  //&#x7B2C;cnt&#x4E2A;&#x72B6;&#x6001;&#x7684;&#x4E8C;&#x8FDB;&#x5236;&#x6570;&#x4E2D;1&#x7684;&#x4E2A;&#x6570;&#x4E3A;sum
    }
    dp[0][1][0] = 1;  //&#x521D;&#x59CB;&#x5316;dp
}

int main()
{
    cin >> n >> k;
    intit();  //&#x8FDB;&#x884C;&#x521D;&#x59CB;&#x5316;
    for(int i=1;i<=n;i++)  //&#x5FAA;&#x73AF;n&#x884C;
        for(int j=1;j<=cnt;j++)  //&#x5FAA;&#x73AF;&#x7B2C;i&#x884C;&#x7684;&#x7B2C;j&#x4E2A;&#x72B6;&#x6001;
            for(int s=0;s<=k;s++)  //&#x524D;i&#x884C;&#x4E00;&#x5171;&#x653E;s&#x4E2A;&#x56FD;&#x738B;
                if(s>=sum1[j]) {  //&#x524D;i&#x884C;&#x653E;&#x7684;&#x56FD;&#x738B;&#x6570;&#x91CF;&#x5F97;&#x8981;&#x4E0D;&#x80FD;&#x5C0F;&#x4E8E;&#x8BE5;&#x884C;&#x8BE5;&#x72B6;&#x6001;&#x7684;&#x56FD;&#x738B;&#x6570;&#x91CF;
                    for(int t=1;t<=cnt;t++)  //&#x904D;&#x5386;&#x4E0A;&#x4E00;&#x884C;&#x6240;&#x6709;&#x72B6;&#x6001;
                        if(!(sta[j]&sta[t])&&!(sta[j]&(sta[t]<<1))&&!(sta[j]&(sta[t]>>1)))
                            dp[i][j][s] += dp[i-1][t][s-sum1[j]];
                }
    ll res = 0;
    for(int j=1;j<=cnt;j++) res+=dp[n][j][k];
    cout << res << endl;
    return 0;
}

玉米田Corn Fields

(题目来源： https://www.luogu.org/problemnew/show/P1879)

题意

给一个m行n列的牧场，然后在一些格子上种草，每一行给出能够种草的格子，不存在任意两个草地有公共边，求方案数。（不种草也是一种方案）

1<=m<=12，1<=n<=12，答案对 100,000,000 取余。

题解

如果需要看这题，请保证自己完全理解上一题并独立完成，因为下面的题解会非常简单。

这一道题多出了一个条件，那就是有一些地方是不能够种草的。那么就不能像上题一样，用一个一维数组sta就能表示所有状态，由于每一行的可行（不存在相邻草地）状态和其数量都是不一样的，那么用二维数组sta和一维计数数组cnt分别记录每一行的状态即可。

其他和上一题基本一样。

状态方程：dp[i][j]为第i行第j个状态的方案数。

代码

#include<bits/stdc++.h>
#define enl cout << "\n"
#define pii pair<ll,int>
#define pdd pair<double , double >
#define pss pair<string , string >
#define pb push_back
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
const ll mod = 100000000;
int m,n,sta[15][5000],grass[15],cnt[15];
ll dp[15][5000];

int main()
{
    cin >> m >> n;
    for(int i=1,a;i<=m;i++) {
        int b = 0;
        //&#x7528;b&#x6765;&#x8868;&#x793A;&#x5F53;&#x524D;&#x884C;&#x54EA;&#x4E2A;&#x4F4D;&#x7F6E;&#x80FD;&#x79CD;&#x8349;&#x54EA;&#x4E2A;&#x4F4D;&#x7F6E;&#x4E0D;&#x80FD;&#xFF0C;1&#x8868;&#x793A;&#x80FD;&#x79CD;&#xFF0C;0&#x8868;&#x793A;&#x4E0D;&#x80FD;
        for(int j=1;j<=n;j++) {
            cin >> a;
            b = ((b<<1)|a);
        }
        grass[i] = b;  //&#x8BB0;&#x5F55;&#x7B2C;i&#x884C;&#x7267;&#x573A;&#x80FD;&#x5426;&#x79CD;&#x8349;&#x7684;&#x60C5;&#x51B5;
    }
    
    //&#x9884;&#x5904;&#x7406;&#x6240;&#x6709;&#x72B6;&#x6001;
    for(int i=0;i<(1<<n);i++) {
        if(i&(i<<1)) continue;  //&#x540C;&#x4E00;&#x884C;&#x76F8;&#x90BB;&#x8349;&#x5730;&#x8DF3;&#x8FC7;
        for(int j=1;j<=m;j++)
            if((i|grass[j]) == grass[j])  //&#x5982;&#x679C;&#x8BE5;&#x884C;&#x79CD;&#x8349;&#x60C5;&#x51B5;&#x6EE1;&#x8DB3;&#x8BE5;&#x884C;&#x7267;&#x573A;&#x7684;&#x60C5;&#x51B5;
                sta[j][++cnt[j]] = i;  //&#x7B2C;j&#x884C;&#x7B2C;cnt[j]&#x4E2A;&#x72B6;&#x6001;&#x7684;&#x4E8C;&#x8FDB;&#x5236;&#x6570;&#x4E3A;i
    }
    
    for(int j=1;j<=cnt[1];j++) dp[1][j] = 1;  //&#x521D;&#x59CB;&#x5316;dp
    for(int i=2;i<=m;i++)  //&#x904D;&#x5386;&#x884C;&#x6570;
        for(int j=1;j<=cnt[i];j++)  //&#x904D;&#x5386;&#x7B2C;i&#x884C;&#x7684;&#x7B2C;j&#x4E2A;&#x72B6;&#x6001;
            for(int t=1;t<=cnt[i-1];t++)  //&#x904D;&#x5386;&#x7B2C;i-1&#x884C;&#x7684;&#x7B2C;t&#x4E2A;&#x72B6;&#x6001;
                if(!(sta[i][j]&sta[i-1][t]))  //&#x5982;&#x679C;&#x6EE1;&#x8DB3;&#x6761;&#x4EF6;
                    dp[i][j] = (dp[i][j] + dp[i-1][t]) % mod;
    ll res = 0;
    for(int j=1;j<=cnt[m];j++) res = (res + dp[m][j]) % mod;
    cout << res << endl;
    return 0;
}

炮兵阵地

(题目来源： https://www.luogu.org/problemnew/show/P2704)

题意

在一个n行m列的地图上，有平原和山地，平原上能装炮台，求炮台之间不能相互攻击时，最多能放多少个炮台。（炮台打击范围请看原题）

1<=n<=100，1<=m<=10

题解

这题刚开始想要稍微贪一点。

于是就定义状态方程：dp[i][j]为第i行第j个状态最多能够放置的炮台数。那么根据题意，该行的状态要受到前两行状态的制约。那么只需要满足下面的条件即可转移：

if(!(sta[i][j]&sta[i-1][s]) && !(sta[i][j]&sta[i-2][t]) && !(sta[i-1][s]&sta[i-2][t]))
    dp[i][j] = max(dp[i][j], dp[i-2][t] + sum1[i-1][s] + sum1[i][j])
//j&#x4E3A;&#x7B2C;i&#x884C;&#x72B6;&#x6001;&#xFF0C;s&#x4E3A;&#x7B2C;i-1&#x884C;&#x72B6;&#x6001;&#xFF0C;t&#x4E3A;&#x7B2C;i-2&#x884C;&#x72B6;&#x6001;

这么看起来似乎没有什么问题，但这个思路在提交的时候没法通过。

这个思路主要的问题在于，第i-1行的状态同样会受到第i-2行和第i-3行的制约！！！

//&#x8FD9;&#x4E2A;&#x610F;&#x601D;&#x662F;&#x8BF4;&#x5982;&#x679C;dp[i][j]&#x7684;&#x662F;&#x7531;dp[i-1][s]&#x548C;dp[i-2][t]&#x5F97;&#x6765;&#x7684;&#xFF0C;&#x4F46;&#x662F;&#xFF01;&#xFF01;&#xFF01;
//&#x5F88;&#x6709;&#x53EF;&#x80FD;dp[i-1][s]&#x662F;&#x7531;dp[i-2][t]&#x548C;dp[i-3][r1]&#x800C;&#x6765;&#x7684;
//&#x800C;dp[i-2][t]&#x662F;&#x7531;dp[i-3][r2]&#x548C;dp[i-4][u]&#x5F97;&#x6765;&#x7684;&#xFF0C;&#x663E;&#x7136;&#x7B2C;i-3&#x884C;&#x7684;&#x72B6;&#x6001;&#x4E0D;&#x4E00;&#x6837;&#xFF01;
//&#x7B80;&#x5355;&#x70B9;&#x5C31;&#x662F;&#x8BF4;&#xFF1A;&#x4E0A;&#x9762;&#x7684;if&#x53EA;&#x5224;&#x65AD;&#x4E86;i,i-1,i-2&#x884C;&#x7684;&#x72B6;&#x6001;&#x662F;&#x53EF;&#x884C;&#x7684;&#xFF0C;&#x4F46;&#x662F;i-1&#x884C;&#x662F;&#x7531;i-3&#x884C;&#x7684;t1&#x72B6;&#x6001;&#x800C;&#x6765;&#x7684;&#xFF0C;i-2&#x884C;&#x662F;&#x7531;i-3&#x884C;&#x7684;t2&#x72B6;&#x6001;&#x800C;&#x6765;&#x7684;&#x8BDD;&#xFF0C;&#x90A3;&#x4E48;&#x4E0A;&#x9762;&#x7684;&#x65B9;&#x5F0F;&#x663E;&#x7136;&#x5C31;&#x662F;&#x4E0D;&#x6210;&#x7ACB;&#x7684;&#xFF01;

根据以上的分析，可以知道，我们不可能一个个从第i行一直往回判断到第1行，所以我们要记录前一行的状态和该状态是由前前一行什么状态而来的。

定义状态方程：dp[i][j][k]为第i行第j个状态且该状态是由上一行k状态而来时，最大放置的炮台数。

至于如何转移看下面的代码。

代码

#include<bits/stdc++.h>
#define enl cout << "\n"
#define pii pair<ll,int>
#define pdd pair<double , double >
#define pss pair<string , string >
#define pb push_back
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
//&#x6CE8;&#x610F;&#x6570;&#x7EC4;&#x7684;&#x5927;&#x5C0F;&#xFF0C;&#x5C0F;&#x5FC3;MLE
int n,m,sta[103][150],cnt[150],sum1[103][150];
ll dp[103][150][150];

int main()
{
    cin >> n >> m;
    for(int i=1;i<=n;i++) {
        string str;
        cin >> str;
        //&#x8FD9;&#x91CC;&#x8DDF;&#x4E0A;&#x4E00;&#x9898;&#x9884;&#x5904;&#x7406;&#x7684;&#x65B9;&#x6CD5;&#x4E00;&#x6837;
        int b = 0;
        for(int j=0;j<m;j++) b = (b<<1)|(str[j]==&apos;P&apos;);
        for(int j=0;j<(1<<m);j++) {
            //&#x5224;&#x65AD;&#x7B2C;i&#x884C;&#x6BCF;&#x4E2A;&#x72B6;&#x6001;j&#x662F;&#x5426;&#x5408;&#x6CD5;
            if((j|b) != b || j&(j<<1) || j&(j<<2)) continue;
            sta[i][++cnt[i]] = j;
            //&#x8BE5;&#x51FD;&#x6570;&#x7528;&#x4E8E;&#x6C42;j&#x7684;&#x4E8C;&#x8FDB;&#x5236;&#x5F62;&#x5F0F;&#x4E2D;&#x6709;&#x51E0;&#x4E2A;1
            sum1[i][cnt[i]] = __builtin_popcount(j);
        }
    }
    
    //&#x521D;&#x59CB;&#x5316;&#x7B2C;1&#x7B2C;2&#x884C;&#x7684;dp&#x503C;
    for(int i=1;i<=cnt[1];i++) dp[1][i][0] = sum1[1][i];
    if(n>1) for(int i=1;i<=cnt[1];i++) for(int j=1;j<=cnt[2];j++) {
        if(!(sta[1][i]&sta[2][j])) dp[2][j][i] = max(dp[2][j][i],dp[1][i][0]+sum1[2][j]);
    }

    for(int i=3;i<=n;i++)  //&#x904D;&#x5386;&#x884C;&#x6570;
        for(int j=1;j<=cnt[i];j++)  //&#x904D;&#x5386;&#x7B2C;i&#x884C;&#x7B2C;j&#x4E2A;&#x72B6;&#x6001;
            for(int s=1;s<=cnt[i-1];s++)  //&#x904D;&#x5386;&#x7B2C;i-1&#x884C;&#x7B2C;s&#x4E2A;&#x72B6;&#x6001;
                if(!(sta[i][j]&sta[i-1][s]))  //&#x82E5;j&#x72B6;&#x6001;&#x4E0E;s&#x72B6;&#x6001;&#x6EE1;&#x8DB3;&#x6761;&#x4EF6;
                    for(int t=1;t<=cnt[i-2];t++)
                        if(!(sta[i][j]&sta[i-2][t]))  //&#x82E5;j&#x72B6;&#x6001;&#x4E0E;t&#x72B6;&#x6001;&#x6EE1;&#x8DB3;&#x6761;&#x4EF6;
                            //&#x8FD9;&#x91CC;&#x53EF;&#x4EE5;&#x4E0D;&#x5224;&#x65AD;t&#x4E0E;s&#x72B6;&#x6001;&#xFF0C;&#x56E0;&#x4E3A;&#x5982;&#x679C;&#x4E0D;&#x6EE1;&#x8DB3;&#xFF0C;&#x90A3;&#x4E48;dp&#x4E3A;0&#x5C31;&#x4E0D;&#x4F1A;&#x4EA7;&#x751F;&#x8D21;&#x732E;
                            dp[i][j][s] = max(dp[i][j][s],dp[i-1][s][t]+sum1[i][j]);
    ll res = 0;
    for(int i=1;i<=cnt[n];i++)
        for(int j=1;j<=cnt[n-1];j++)
            res = max(res,dp[n][i][j]);
    cout << res << endl;
    return 0;
}
