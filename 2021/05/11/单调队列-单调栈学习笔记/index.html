单调队列

题目：https://www.luogu.com.cn/problem/P1886

解析

8 3
1 3 -1 -3 5 3 6 7

直接模拟一遍过程比较直观，先以求最大值为例。

定义(dat,pos)为第pos个位置，元素值为dat。

1.input:1 and deque:(1,1)

2.input:3 and deque:(3,2)

我们要时刻保证队列是绝对单调的，并且单调递减，保证队首元素为区间最大值。

所以从队尾踢出(1,1)再入队(3,2)。

3.input:-1 and deque:(3,2),(-1,3)

4.input:-3 and deque:(3,2),(-1,3),(-3,4)

5.input:5 and deque:(5,5)

该步5比队列中每个数都要大，所以踢出队列中所有数。

但是在从队尾把比5小的数踢出之前，要先把不在区间中的数从队首踢出。

也就是说该步把(3,2)从队首踢出，然后把剩下两个数从队尾踢出在入队(5,5)

6.input:3 and deque:(5,5),(3,6)

7:input:6 and deque:(6,7)

8:input:7 and deque:(7,8)

上述每一步的队首元素就是当前区间的最大值，求最小值同理。

这里可以思考一下这种方法的正确性，首先每一步队内元素必然都是处于当前区间中的，并且pos值从小到大，dat值从大到小，我们把一个新输入的数入队时，把比其小的数都出队了，这个并不会影响正确性，因为往后扫的时候队首元素踢出后，最大值的位置也是先轮到该输入的值，直到当前输入的值被踢出后，也轮不到这些出队的小值当老大。（仔细思考，非常简单）

代码

#include<bits/stdc++.h>
#define enl cout << "\n"
#define jud cout << "caonima" << endl;
#define pii pair<int,int>
#define pll pair<long long,long long>
#define pdd pair<double , double >
#define pss pair<string , string >
#define pb push_back
#define Clear(x) while(!x.empty())x.pop()
#define pqg(type) priority_queue<type,vector<type>,greater<type>>
#define pql(type) priority_queue<type,vector<type>,less<type>>
#define Max(a,b) ((a)<(b)?(b):(a))
#define Min(a,b) ((a)<(b)?(a):(b))
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
const int maxn = 1e6 + 3;
struct Node{
    int dat,pos;
    Node(int dat,int pos):dat(dat),pos(pos) { }
};
int t,n,k;
deque<Node > Max,Min;
vector<int > resmax,resmin;

void solve()
{
    cin >> n >> k;
    for(int i=1,tem;i<=n;i++) {
        cin >> tem;
        Node nod(tem,i);
        while(!Max.empty() && Max.front().pos<=i-k) Max.pop_front();
        while(!Max.empty() && Max.back().dat<=nod.dat) Max.pop_back();
        Max.pb(nod);
        while(!Min.empty() && Min.front().pos<=i-k) Min.pop_front();
        while(!Min.empty() && Min.back().dat>=nod.dat) Min.pop_back();
        Min.pb(nod);
        if(i>=k) resmax.pb(Max.front().dat);
        if(i>=k) resmin.pb(Min.front().dat);   
    }
    for(int i=0;i<resmin.size();i++) cout << resmin[i] << " "; enl;
    for(int i=0;i<resmax.size();i++) cout << resmax[i] << " "; enl;
}

int main()
{
    //freopen("data.in","r",stdin);
    ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    //cin >> t;
    t = 1;
    while(t--)
        solve();
    return 0;
}

单调栈

题目：https://www.luogu.com.cn/problem/P5788

理解了单调队列后，单调栈也比较好理解了。单调栈主要目的还是求第i个数左边或右边第一个比其大或小的数的所在下标。

5
1 4 2 3 5

还是模拟一遍，这次就没有注释了。

1.input:1 and stack:(1,1)

2.input:4 and stack:(4,2)

3.input:2 and stack:(4,2),(2,3)

4.input:3 and stack:(4,2),(3,4)

5.input:5 and stack:(5,5)

在每次入队新元素之前，先把栈中比其小的元素出栈并更新f[stack.top_pos] = newinput_pos。

代码

#include<bits/stdc++.h>
#define enl cout << "\n"
#define jud cout << "caonima" << endl;
#define pii pair<int,int>
#define pll pair<long long,long long>
#define pdd pair<double , double >
#define pss pair<string , string >
#define pb push_back
#define Clear(x) while(!x.empty())x.pop()
#define pqg(type) priority_queue<type,vector<type>,greater<type>>
#define pql(type) priority_queue<type,vector<type>,less<type>>
#define Max(a,b) ((a)<(b)?(b):(a))
#define Min(a,b) ((a)<(b)?(a):(b))
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
const int maxn = 3e6 + 3;
struct Node{
    int dat,pos;
    Node(int dat,int pos):dat(dat),pos(pos) { }
};
int t,n,f[maxn];
stack<Node > st;

void solve()
{
    cin >> n;
    for(int i=1,tem;i<=n;i++)
    {
        cin >> tem;
        while(!st.empty() && st.top().dat<tem) f[st.top().pos] = i, st.pop();
        st.push(Node(tem,i));
    }
    while(!st.empty()) f[st.top().pos] = 0, st.pop();
    for(int i=1;i<=n;i++) cout << f[i] << " ";
    enl;
}

int main()
{
    //freopen("data.in","r",stdin);
    ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    //cin >> t;
    t = 1;
    while(t--)
        solve();
    return 0;
}
