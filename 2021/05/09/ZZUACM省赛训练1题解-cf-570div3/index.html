A - Nearest Interesting Number

题意

给一个正整数a<=1000，找一个数n>=a使得n各个位数之和被4整除。

题解

枚举符合条件的n，直接验证各个位数数字之和是否被4整除即可。

代码

#include<bits/stdc++.h>
#define enl cout << "\n"
#define pii pair<int,int>
#define pdd pair<double , double >
#define pss pair<string , string >
#define pb push_back
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
int a;

int main()
{
    ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    cin >> a;
    for(int i=a;;i++) {
        int res = 0,j = i;
        while(j) res+=j%10, j/=10;
        if(res%4==0) {
            cout << i << endl;
            break;
        }
    }
    return 0;
}

B - Equalize Prices

题意

q个询问，每个询问给出n，k和n个数。

找一个尽可能大的数B使得|ai-B|<=k。

1<=q<=100,1<=n<=100,1<=k<=1e8,1<=ai<=1e8

题解

对于每一个数ai，B只能在[ai-k,ai+k]之间取值。则每个数ai都有一个区间，求所有区间交集，然后取右端点即可。

代码

#include<bits/stdc++.h>
#define enl cout << "\n"
#define pii pair<int,int>
#define pdd pair<double , double >
#define pss pair<string , string >
#define pb push_back
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
int t,n,k;
vector<pii > v;

int main()
{
    ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    cin >> t;
    while(t--)
    {
        v.clear();
        cin >> n >> k;
        for(int i=1,a;i<=n;i++) {
            cin >> a;
            v.pb(pii(a-k,a+k));
        }
        pii res(v[0].first,v[0].second);
        int ok = 1;
        for(int i=1;i<v.size();i++) {
            if(res.second < v[i].first || v[i].second < res.first) {
                cout << -1 << endl;
                ok = 0;
                break;
            }
            if(v[i].first >= res.first) res = pii(v[i].first,min(res.second,v[i].second));
            else if(v[i].second <= res.second) res = pii(max(v[i].first,res.first),v[i].second);
        }
        if(ok) cout << res.second << endl;
    }
    return 0;
}

C - Computer Game

题意

q个询问，每个询问给出k，n，a，b。一台电脑有k格电量，小明想玩n轮游戏，当电脑剩余电量大于a时，能玩a游戏，同时电量减少a。当电脑剩余电量大于b时，能玩b游戏，同时电量减少b。求是否能进行n轮游戏（n轮游戏后剩余电量大于0），不能输出’-1’，如果能，尽可能多地玩a游戏，并输出玩a游戏的轮数。

1<=q<=1e5,1<=k,n<=1e9,1<=b<a<=1e9

题解

设进行了x轮a游戏，y轮b游戏。

得出不等式：k-ax-by>0，x+y>=n。

（关于不等式1的正确性：仔细分析是不是感觉少了k-ax>a，k-by>b之类的限制条件，但实际上，k-ax-a>0，即k-a(x+1)>0，此时y=0，所以我们不需要但是进行x轮游戏后电量是否还大于a诸如此类的问题，因为第一个不等式就已经包含了）

通过不等代换，解得：$ x < \frac{k-bn}{a-b} $

（验证右式是否为小数，根据验证输出相应的答案）

代码

#include<bits/stdc++.h>
#define enl cout << "\n"
#define pii pair<int,int>
#define pdd pair<double , double >
#define pss pair<string , string >
#define pb push_back
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
ll t,n,k,a,b;

int main()
{
    ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    cin >> t;
    while(t--)
    {
        cin >> k >> n >> a >> b;
        ll A = k-b*n, B = a-b;
        if(A<=0) cout << -1 << endl;
        else {
            if(A%B == 0) cout << min(n,(A/B-1)) << endl;
            else cout << min(n,(A/B)) << endl;
        }
    }
    return 0;
}

D - Candy Box (easy version)

题意

给出q个询问，每个询问给出n与n个数，表示有n种糖果，这n个数为糖果的类型编号，现在要选一些糖果作为礼物送出去，如果这个礼物中有两种糖果的数量一样，那么这个礼物就不是好礼物。在好礼物的条件下，求礼物中最多能够多少个糖果。

1<=q<=2e5,1<=n<=2e5,1<=ai<=n,all(n)<=2e5

题解

把每种糖果的数量求出来，并将这些数量从大到小排序，然后依次选择。

比如3，3，2，那么就可以选3，2，1

比如4，4，4，4，4，4，那就可以选4，3，2，1

比如10，3，3，那就可以选10，3，2

代码

#include<bits/stdc++.h>
#define enl cout << "\n"
#define pii pair<int,int>
#define pdd pair<double , double >
#define pss pair<string , string >
#define pb push_back
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
const int maxn = 2e5 + 3;
int t,n,a[maxn];
priority_queue<int , vector<int > , less<int > > pq;

int main()
{
    ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    cin >> t;
    while(t--)
    {
        memset(a,0,sizeof(0));
        while(!pq.empty()) pq.pop();
        cin >> n;
        for(int i=1;i<=n;i++) cin >> a[i];
        sort(a+1,a+n+1);
        a[0] = a[n+1] = 0;
        int tot = 0, vi = 0;
        for(int i=1;i<=n+1;i++) {
            if(a[i] != a[i-1]) {
                if(i!=1) pq.push(tot);
                tot = 1;
            }else ++tot;
        }
        ll res = 0, fron = pq.top()+1;  //fron&#x8BB0;&#x5F55;&#x524D;&#x9762;&#x90A3;&#x79CD;&#x7CD6;&#x679C;&#x9009;&#x4E86;&#x51E0;&#x4E2A;
        while(!pq.empty()) {
            if(fron == 1) break;
            //&#x5982;&#x679C;&#x5F53;&#x524D;&#x79CD;&#x7C7B;&#x7684;&#x7CD6;&#x679C;&#x591F;&#x6570;&#xFF0C;&#x90A3;&#x4E48;&#x9009;fron-1&#x4E2A;&#xFF0C;&#x4E0D;&#x591F;&#x6570;&#x5C31;&#x9009;&#x5F53;&#x524D;&#x79CD;&#x7C7B;&#x7684;&#x7CD6;&#x679C;&#x603B;&#x6570;
            if(pq.top() >= fron) res+=fron-1, pq.pop(), fron-=1;
            else res+=pq.top(), fron = pq.top(), pq.pop();
        }
        cout << res << endl;
    }
    return 0;
}

E - Subsequences (easy version)

题意

给出n，k，和一个长度为n的字符序列。选出k个不同的子序列放入集合中，如果该子序列长度为t，那么放入集合需要花费n-t的单位的价格。如果不能选出k个不同的子序列，输出’-1’，如果能，那么输出最小的花费。

子序列：”asdf”的子序列有”asdf”,”adf”,”df”,””等等。

1<=n,k<=100

题解

刚开始有一个比较笨的想法：当子序列长度为n时，有一种。长度为n-1时，分别循环删除那个字符，有n种，然后长度为n-2时，遍历删除哪两个不同的字符，如此下去。显然这种方法难以实现，如果要删掉10个字符，可能就得要11个字符串相连，比较麻烦。

明显可以注意到，n-k长度的子序列可以由n-k+1长度的子序列得到，这样每一轮对于上一轮的每一个字符串只需要删掉其中一个字符即可。这样遍历次数和上面那种方法一样，但其实k<=100，很快就可以得出答案，不需要遍历所有子序列。所以这个优化主要还要解决字符串拼接困难的问题。

代码

#include<bits/stdc++.h>
#define enl cout << "\n"
#define judge cout << "caonima" << endl;
#define pii pair<ll,int>
#define pdd pair<double , double >
#define pss pair<string , string >
#define pb push_back
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
int n,k,res;
string str;
set<string > se,temse;

int main()
{
    //freopen("data.in","r",stdin);
    ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    cin >> n >> k >> str;
    --k; se.insert(str);
    //se&#xFF0C;temse&#x53EF;&#x4EE5;&#x7528;&#x4E8E;&#x53BB;&#x91CD;
    for(;k>0&&!se.empty();)
    {
        while(!se.empty()) {
            //&#x5BF9;&#x4E8E;&#x4E0A;&#x4E00;&#x8F6E;&#x751F;&#x6210;&#x7684;&#x6BCF;&#x4E00;&#x4E2A;&#x5B57;&#x7B26;&#x4E32;
            string s = *se.begin(), t; se.erase(*se.begin());
            //&#x4F9D;&#x6B21;&#x5220;&#x6389;&#x4E00;&#x4E2A;&#x5B57;&#x7B26;&#x540E;&#x62FC;&#x63A5;&#x5728;&#x4E00;&#x8D77;
            for(int i=0;i<s.length();i++) {
                t = s.substr(0,i) + s.substr(i+1,s.length()-i-1);
                temse.insert(t);
                if(temse.size() == k) break;
            }
            if(temse.size() == k) break;
        }
        if(temse.size() == k) break;
        while(!temse.empty())
        {
            se.insert(*temse.begin());
            res += str.length() - (*temse.begin()).length();
            temse.erase(*temse.begin());
            --k;
        }
    }
    while(!temse.empty()) res += str.length() - (*temse.begin()).length(), temse.erase(*temse.begin()), --k;
    if(k == 0) cout << res << endl;
    else cout << -1 << endl;
    return 0;
}

F - Topforces Strikes Back

题意

q个询问，每个询问给出n与n个数，在这n个数中至多选出3个数，并保证选出来的数不能互相整除，求选出来的数之和的最大值并输出。

1<=q<=2e5,1<=n<=2e5,2<=ai<=2e5,all(n)<=2e5

题解

玄学暴力枚举题。（复杂度难以证明或观察，玄学过题就可以了）

本题比较有意义的地方在于一些比较玄学的剪枝方法。

基本的枚举思路时，将所有数排序，然后贪心，从最大的开始选，然后往小的找，发现没有整除关系，然后继续找第三个数，如果发现更好的答案后更新，具体看代码。

代码

#include<bits/stdc++.h>
#define enl cout << "\n"
#define judge cout << "caonima" << endl;
#define pii pair<ll,int>
#define pdd pair<double , double >
#define pss pair<string , string >
#define pb push_back
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
const int maxn = 2e5 + 3;
int t,n;
set<int > se;
vector<int > v;

int main()
{
    ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    cin >> t;
    while(t--)
    {
        cin >> n;
        se.clear(); v.clear();
        //&#x526A;&#x679D;&#x4E00;&#xFF1A;&#x53BB;&#x91CD;&#xFF0C;&#x76F8;&#x540C;&#x7684;&#x6570;&#x6BD4;&#x4E0D;&#x53EF;&#x80FD;&#x9009;&#xFF0C;&#x6240;&#x4EE5;&#x53EA;&#x8003;&#x8651;&#x4E00;&#x4E2A;&#x5373;&#x53EF;
        for(int i=1,x;i<=n;i++) cin >> x, se.insert(x);
        for(auto it: se) v.pb(it);
        //&#x526A;&#x679D;&#x4E8C;&#xFF1A;&#x5BF9;&#x4E8E;&#x53EA;&#x9009;&#x4E00;&#x4E2A;&#x6570;&#xFF0C;&#x90A3;&#x4E48;&#x9009;&#x6700;&#x5927;&#x5C31;&#x662F;&#x6700;&#x597D;&#x7684;&#xFF0C;&#x540E;&#x9762;&#x5BF9;&#x4E8E;&#x4E00;&#x4E2A;&#x6570;&#x65F6;&#x5C31;&#x4E0D;&#x7528;&#x66F4;&#x65B0;res=max(res,v[i])&#x4E86;
        int res = v[v.size()-1];
        for(int i=v.size()-1;i>=0;i--) {
            for(int j=i-1;j>=0;j--) if(v[i]%v[j]) {
                res = max(res,v[i]+v[j]);
                //&#x526A;&#x679D;&#x4E09;&#xFF1A;j=0&#x65F6;&#x4E0D;&#x7528;&#x627E;k&#x4E86;&#xFF0C;v[i]+v[j]+v[j-1]&#x5982;&#x679C;&#x5C0F;&#x4E8E;res&#x4E5F;&#x4E0D;&#x7528;&#x627E;k&#x4E86;
                if(j==0 || v[i] + v[j] + v[j-1] <= res) break;
                for(int k=j-1;k>=0;k--) {
                    //&#x526A;&#x679D;&#x56DB;&#xFF1A;&#x8DDF;&#x4E09;&#x5DEE;&#x4E0D;&#x591A;
                    if(v[i]+v[j]+v[k] <= res) break;
                    if(v[i]%v[k] && v[j]%v[k]) res = v[i]+v[j]+v[k];
                }
            }
        }
        cout << res << endl;
    }   
    return 0;
}
