1001 Mod, Or and Everything

题意

T组数据，给一个n，求((n mod 1) or (n mod 2) or … or (n mod n))

T<=5000, 1<=n<=1e12

题解

打表找规律

附上证明：明显可以发现，$ (n\ mod\ n) $ 一直到 $ (n\ mod\ \lceil\frac{n}{2}\rceil+1 ) $，结果为$ 0,1…\lceil\frac{n}{2}\rceil-1 $，由于是或运算，明显可以得到答案。

代码

#include<bits/stdc++.h>
typedef long long ll;
using namespace std;
int T;
ll n;

ll fastpow(ll a, ll b) { ll res=1;while(b){if(b&1)res*=a;b>>=1;a*=a;} return res; }

void solve()
{
    cin >> n;
    ll dig = 0, tot1 = 0, nn = n;
    while(nn) tot1 += nn&1, ++dig, nn>>=1;
    if(n == 1 || n == 2) cout << 0 << endl;
    else if(tot1 == 1) cout << (fastpow(1ll<<1, dig - 2) - 1) << endl;
    else cout << (fastpow(1ll<<1, dig - 1) - 1) << endl;
}

int main()
{
    cin >> T;
    while(T--)
        solve();
    return 0;
}

1005 Minimum spanning tree

题意

T组数据，给一个n，有n-1个点，编号为2,3,4,…,n，a点与b点有一条无向边相连，边权为lcm(a,b)，求该图最小生成树

T<=100，2<=n<=1e7

题解

数学观察题

考虑每一个数的贡献，显然合数的最小贡献为本身，质数的最小贡献为本身的2倍（注意2不产生贡献）

处理：先求出所有素数然后求前缀和，每次等差数列求和后，二分找到素数的前缀和，相加即可

代码

#include<bits/stdc++.h>
#define pb push_back
typedef long long ll;
using namespace std;
const int maxn = 1e7 + 3;
int T;
ll n;
int vis[maxn];
ll ans[670000];
vector<ll > prim;

void getprim()
{
    int nn = maxn-3;
    for(int i=2;i<=nn;i++) {
        if(!vis[i]) prim.pb(i);
        for(int j=0;j<prim.size() && i*prim[j]<=nn;j++) {
            vis[i*prim[j]] = 1;
            if(i%prim[j] == 0) break;
        }
    }
}

void intit()
{
    getprim();
    for(int i=1;i<=prim.size();i++) ans[i] = ans[i-1] + prim[i];
}

void solve()
{
    cin >> n;
    ll res = n*(n+1)/2 - 3;
    int pos = lower_bound(prim.begin(), prim.end(), n) - prim.begin();
    if(pos == prim.size() || prim[pos] != n) cout << res+ans[pos-1] << endl;
    else cout << res+ans[pos] << endl;
}

int main()
{
    intit();
    cin >> T;
    while(T--)
        solve();
    return 0;
}

1006 Xor sum

题意

T组数据，给定长度为n的序列，求最短连续子序列，使得该序列的异或和不小于k，输出该序列的左右区间，若有多种答案，输出左端点最小的那一种，无解输出-1

T<=100,1<=n<=1e5,0<=k<2^30,0<=ai<2^30,不超过5个样例n>1000

题解

区间异或和问题，异或前缀和+01字典树

先求异或前缀和，问题转化为在a[0]到a[n]中找到两个不同的数使得异或小于k

从左到右枚举右端点，在字典树中找到符合条件的最大左端点，字典树中存的是对应数的最大下标

代码

#include<bits/stdc++.h>
using namespace std;
const int maxn = 1e5 + 3;
int T, n, k, a[maxn];
int tr[3000003][3], root = 0, cnt = 0;

void insert(int rot, int num, int pos)
{
    for(int i=29;i>=0;i--) {
        int dig = num >> i & 1;
        if(tr[rot][dig]) tr[tr[rot][dig]][2] = pos;
        else {
            tr[rot][dig] = ++cnt;
            tr[cnt][2] = pos;
            tr[cnt][0] = tr[cnt][1] = 0;
        }
        rot = tr[rot][dig];
    }
}

int query(int rot, int num)
{
    int res = -1;
    for(int i=29;i>=0;i--) {
        int diga = num >> i & 1, digk = k >> i & 1;
        if(!digk) {
            if(tr[rot][diga^1]) res = max(res, tr[tr[rot][diga^1]][2]);
            if(!tr[rot][diga]) return res;
            rot = tr[rot][diga];
        }else {
            if(!tr[rot][diga^1]) return res;
            rot = tr[rot][diga^1];
        }
        if(i == 0) res = max(res, tr[rot][2]);
    }
    return res;
}

void solve()
{
    cnt = 0; tr[root][0] = tr[root][1] = 0;
    int ok = 0;
    cin >> n >> k;
    for(int i=1;i<=n;i++) {
        cin >> a[i];
        if(a[i] >= k && ok == 0) ok = i;
    }
    if(ok) {
        cout << ok << " " << ok << endl;
        return ;
    }

    for(int i=1;i<=n;i++) a[i] = a[i-1]^a[i];
    int resl = -1, resr = -1;
    insert(root, 0, 0);
    for(int i=1;i<=n;i++) {
        int right = query(root, a[i]);
        if(right != -1 && (resl == -1 || i-right < resr-resl || (i-right==resr-resl&&right<resl))) resl = right, resr = i;
        insert(root, a[i], i);
    }

    if(resl == -1) cout << -1 << endl;
    else cout << resl+1 << " " << resr << endl;
}

int main()
{
    ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    cin >> T;
    while(T--)
        solve();
    return 0;
}

总结：经典题目了，之前似乎有遇到过，但是没有想起来，比赛中的状态需要实时调整，不要被其他东西所影响了。除此之外要加强各种题型的练习。这题字典树数组至少要200万，开小了会tle，可能也会wa，就挺奇怪的，以后算不出数组大小要尽量大。这题另一个比较奇怪的地方就是，数组开200万和300万，亦或是1000万都是同一个内存大小，真特么奇怪了。

1008

题意

T组数据，给一个n*m的矩阵，求最大子矩阵（面积）使得子矩阵中每一列从上往下非严格递增

T<=10,1<=n,m<=2000,1<=vij<=5000,不超过2个样例使得n*m>1000

题解

悬线法，单调栈

先O(nm)处理第i行第j列往上最多能到多远，存在res数组中。

例如对于某一行，res为2 2 1 3 2，最大子矩阵为5，若res为3 2 3 2 3，最大子矩阵为10

对于res的第i行第j个数，求出向左最远能到多少，向右最远能到多少，然后更新答案。

代码

#include<bits/stdc++.h>
#define pii pair<int,int>
using namespace std;
const int maxn = 2003;
int T, n, m, mat[maxn][maxn], res[maxn][maxn], l[maxn], r[maxn];
stack<pii > s;

void solve()
{
    cin >> n >> m;
    for(int i=1;i<=n;i++) for(int j=1;j<=m;j++) cin >> mat[i][j];
    for(int j=1;j<=m;j++) {
        int cnt = 0;
        for(int i=1;i<=n;i++) {
            if(mat[i][j] >= mat[i-1][j]) ++cnt;
            else cnt = 1;
            res[i][j] = cnt;
        }
    }

    int area = m;
    for(int i=2;i<=n;i++) {
        for(int j=1;j<=m;j++) {
            //l&#x4E2D;&#x5B58;&#x7684;&#x662F;&#x5411;&#x5DE6;&#x6700;&#x8FDC;&#x80FD;&#x5230;&#x591A;&#x8FDC;&#xFF0C;r&#x4E2D;&#x5B58;&#x7684;&#x662F;&#x5411;&#x53F3;&#x6700;&#x8FDC;&#x80FD;&#x5230;&#x591A;&#x8FDC;
            while(!s.empty() && s.top().first > res[i][j]) r[s.top().second] = j-1, s.pop();
            s.push(pii(res[i][j], j));
        }
        while(!s.empty()) r[s.top().second] = m, s.pop();
        for(int j=m;j>=1;j--) {
            while(!s.empty() && s.top().first > res[i][j]) l[s.top().second] = j+1, s.pop();
            s.push(pii(res[i][j], j));
        }
        while(!s.empty()) l[s.top().second] = 1, s.pop();
        for(int j=1;j<=m;j++) area = max(area, res[i][j]*(r[j]-l[j]+1));
    }

    cout << area << endl;
}

int main()
{
    ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    cin >> T;
    while(T--)
        solve();
    return 0;
}

1009 KD-Graph

题意

给出n个点，m条边的有权无向图，n个点按照以下规定分成K组：

1.若p点与q点存在于一组，则至少有一条p到q的路径，路径上的最大边小于等于D；

2.若p点与q点不存在于一组，则必没有一条p到q的路径，路径上的最大边小于等于D；

给出n,m,k找出最小的非负整数D，使得图G恰好能被分成k组，若不存在输出-1

T<=5,2<=n<=1e5,1<=m<=5e5,1<=k<=n,1<=u,v<=n,u!=v,1<=w<=1e9

题解

没看出来这是一道简单题。

明显可以知道答案必然是某一条边的权值。

那么从小到大排序边，这里第一个需要思考的地方就是如果快速对其分组，仔细思考可以发现只需要对每一条边进行操作，若改边小于等于D那么改边的两点同属于一个集合，而且具有传递性。

同时又可以发现，从小到大遍历边，然后把点并进一个个集合，每一次操作最多使得组数减少1，那么只需要从小到大遍历边，然后用并查集把点合并，若找到正好能够分成k组即找到答案。注意，我们每次必须要同时把权值相同的边一起合并了才能判断 。

代码

#include<bits/stdc++.h>
using namespace std;
const int maxn = 1e6 + 3, maxm = 1e6 + 3;
int T, n, m, k, f[maxn];
struct Edge{
    int u, v, w;
}e[maxm]; 
bool cmp(struct Edge a, struct Edge b) { return a.w < b.w; }

int getf(int v) { return f[v]==v?v:f[v]=getf(f[v]); }

void solve()
{
    cin >> n >> m >> k;
    for(int i=1;i<=m;i++) cin >> e[i].u >> e[i].v >> e[i].w;
    sort(e + 1, e + m + 1, cmp);
    for(int i=1;i<=n;i++) f[i] = i;
    int tot = n, i = 1; e[m+1].w = e[m].w + 1;
    //&#x6CE8;&#x610F;&#x8FD9;&#x91CC;&#x4E0D;&#x8981;&#x5224;&#x65AD;k==1&#x65F6;&#x8F93;&#x51FA;e[m].w
    //&#x56E0;&#x4E3A;2 2 2 3 3&#xFF0C;&#x53EF;&#x80FD;&#x524D;&#x4E09;&#x6761;&#x8FB9;&#x5C31;&#x80FD;&#x591F;&#x628A;&#x6240;&#x6709;&#x70B9;&#x5206;&#x6210;1&#x7EC4;&#x4E86;&#xFF0C;&#x7B54;&#x6848;&#x5219;&#x662F;2&#x4E0D;&#x662F;3
    for(int i=1;i<=m+1;i++) {
        //&#x628A;&#x524D;&#x4E00;&#x6BB5;&#x6743;&#x503C;&#x76F8;&#x540C;&#x7684;&#x8FB9;&#x5904;&#x7406;&#x4E86;&#x4E4B;&#x540E;&#x518D;&#x5224;&#x65AD;tot==k&#x6765;&#x770B;&#x770B;&#x662F;&#x5426;&#x80FD;&#x591F;&#x5206;&#x6210;k&#x7EC4;&#x4E86;
        if(e[i].w != e[i-1].w && tot == k) {
            cout << e[i-1].w << endl;
            return ;
        }
        int t1 = getf(f[e[i].u]), t2 = getf(f[e[i].v]);
        if(t1 == t2) continue;
        --tot;
        f[t2] = t1;
        //&#x7EC4;&#x6570;&#x662F;&#x9012;&#x51CF;&#x7684;&#xFF0C;&#x53EA;&#x8981;tot<k&#x90A3;&#x5C31;&#x4E0D;&#x4F1A;&#x5B58;&#x5728;&#x7B54;&#x6848;
        if(tot < k) break;
    }
    cout << -1 << endl;
}

int main()
{
    ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    cin >> T;
    while(T--)
        solve();
    return 0;
}
