B Black and white

题意

有一个n*m的矩阵，所有格子都是白色的，第i行第j列的费用为cij，将该格子涂成黑色需要的费用为cij。任选连续两行与连续两列，中间交有4个格子，如果该4个格子中有3个已经是黑色，那么剩下一个格子涂成黑色则不需要任何费用，求将所有格子涂成黑色的最小花费。

所有格子的费用由如下方式生成，（有a,b,c,d,p）

A0 = a

Ai+1 = ( Ai * Ai * b + Ai * c + d ) % p

假设对于一个3*3的矩阵，其各个格子的费用为：

A1 A2 A3

A4 A5 A6

A7 A8 A9

（其余类型的矩阵由类似以上方式生成）

题解

最小生成树模板。

该题的转化非常巧妙，在纸上多画几张图仔细思考其实就可以发现，把行号当成n个点，把列号当成m个点，原问题可以转化为对于一个有n+m个点的图，求最小生成树，行点i和列点j的权值为G[i][j]。

（稠密图用prim，稀疏图用kruskal）

本题用prim（代码的关键在于lowset数组，O(n^2)复杂度，具体可以回去复习以下。kruskal的关键在于并查集）

代码

#include<bits/stdc++.h>
typedef long long ll;
using namespace std;
const int maxn = 5003;
const ll inf = 0x3f3f3f3f3f3f3f3f;
ll n,m,a,b,c,d,p,mat[maxn][maxn];
ll lowr[maxn], lowc[maxn];

int main()
{
    ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    cin >> n >> m >> a >> b >> c >> d >> p;
    ll num = a;
    for(int i=1;i<=n;i++) for(int j=1;j<=m;j++) {
        num = (num*num*b%p + num*c%p + d) % p;
        mat[i][j] = num;
    }
    //lowset&#x6570;&#x7EC4;&#xFF0C;&#x5206;&#x522B;&#x8868;&#x793A;&#x884C;&#x70B9;&#x548C;&#x5217;&#x70B9;
    memset(lowr, 0x3f, sizeof lowr);
    memset(lowc, 0x3f, sizeof lowc);
    ll res = 0;
    //&#x8FD9;&#x91CC;&#x4E0D;&#x80FD;&#x7528;0&#x8868;&#x793A;&#x5DF2;&#x7ECF;&#x5728;&#x6811;&#x4E2D;&#xFF0C;&#x8981;&#x7528;-1&#xFF0C;&#x56E0;&#x4E3A;&#x8FB9;&#x6743;&#x53EF;&#x80FD;&#x4E3A;0
    lowr[1] = -1; for(int j=1;j<=m;j++) lowc[j] = mat[1][j];
    for(int k=1;k<=n+m-1;k++) {
        int state, tempi; ll temp = inf;
        for(int i=1;i<=n;i++) if(lowr[i]!=-1 && lowr[i]<temp) temp=lowr[i], tempi=i, state = 0;
        for(int j=1;j<=m;j++) if(lowc[j]!=-1 && lowc[j]<temp) temp=lowc[j], tempi=j, state = 1;
        if(state == 0) {
            res += temp;
            lowr[tempi] = -1;
            for(int j=1;j<=m;j++) lowc[j] = min(lowc[j], mat[tempi][j]);
        }
        if(state == 1) {
            res += temp;
            lowc[tempi] = -1;
            for(int i=1;i<=n;i++) lowr[i] = min(lowr[i], mat[i][tempi]);
        }
    }
    
    cout << res << endl;
    return 0;
}

E Math

题意

T组数据，每组给一个n求有多少对(x,y)满足：

$$ 1 \leq x \leq y \leq n, 并且xy+1|x^{2}+y^{2}$$

T<=1e5, 1<=n<=1e18

题解

据说是IMO原题。（1988年第6题？？？）

可以打表找规律（我这个废物应该是没有心情看规律了，签到题都没做出来）

这道纯数学题涉及的技巧还蛮多的，可以看看下面的证明：

$ 设 x^2+y^2 = k(xy+1) \Rightarrow x^2+y^2-kxy-k = 0 $

$ 固定x,k, 把上式看成关于y的一元二次方程, 由韦达定理 $

$ y + y’ = kx $

$ yy’ = x^2-k $

$ 假设存在一个解(x, y)满足原式(由于对称性，不妨设x \leq y)$

$ 那么(x, y’)也是一个解, 并且y’<y, 同时y’ < x $

$ （附证：yy’ = x^2-k \leq y^2-k \Rightarrow y’ \leq y - \frac{k}{y} \Rightarrow y’ < y ）$

$ （附证：yy’ = x^2-k \geq xy’ \Rightarrow y’ \leq x - \frac{k}{x} \Rightarrow y’ < x） $

$ 由上述的韦达定理可以得出y’ = kx - y $

$ 那么由解(x, y)就能够导出另外一个解(kx-y, x) $

$ 可以发现解是不断减小的，最多可以减小到0的（不要对应题目条件），那么有： $

$ 0 = x^2 - k \Rightarrow k = x^2 $

$ 这样我们只需要找到最小的解就能够倒推出更大的解 $

$ 通过观察（其实是打表）可以发现(x, x^3)是一种解 $

$ 那么向上倒推，(x^3, x^5-x)也是一个解，那就可以得到所有类型的解了。 $

注意到x^3最多能够取到1e6，预处理然后二分查询，所以复杂度是O(1e6+Tlog(1e6))

代码

#include<bits/stdc++.h>
#define pb push_back
#define pll pair<unsigned long long,unsigned long long>
typedef unsigned long long ull;
using namespace std;
ull T, n;
vector<pll > v;

void intit()
{
    //&#x9884;&#x5904;&#x7406;&#x7B2C;&#x4E00;&#x79CD;&#x7C7B;&#x578B;&#x7684;&#x89E3;&#xFF0C;&#x7531;&#x4E8E;y&#x662F;&#x5404;&#x4E0D;&#x76F8;&#x540C;&#x7684;&#xFF0C;&#x65B9;&#x4FBF;&#x67E5;&#x8BE2;&#xFF0C;vector&#x91CC;&#x5B58;&#x7684;&#x662F;(y,x)
    for(ull x = 1;x*x*x <= 1e18; x++) v.pb(pll(x*x*x,x));
    int end = v.size();
    for(int i=1;i<end;i++) {
        ull x = v[i].second, y = v[i].first;
        ull k = x*x;
        //&#x5BF9;&#x4E8E;&#x7B2C;&#x4E00;&#x7C7B;&#x578B;&#x7684;&#x6240;&#x6709;&#x89E3;&#xFF0C;&#x6784;&#x9020;&#x51FA;&#x6240;&#x6709;&#x7B26;&#x5408;&#x65B9;&#x7A0B;&#x7684;&#x89E3;
        while(1) {
            //&#x7531;&#x4E8E;&#x6570;&#x636E;&#x8303;&#x56F4;&#xFF0C;&#x8FD9;&#x91CC;&#x7528;&#x9664;&#x6CD5;&#x6765;&#x5224;&#x65AD;&#x662F;&#x5426;&#x6EA2;&#x51FA;
            if(y > (x+1e18)/k+1) break;
            ull temp = x; x = y; y = k * y - temp;
            v.pb(pll(y, x));
        }
    }
    sort(v.begin(), v.end());
}

int main()
{
    intit();
    cin >> T;
    while(T--) {
        cin >> n;
        int pos = lower_bound(v.begin(), v.end(), pll(n,n)) - v.begin();
        //&#x907F;&#x514D;&#x8D8A;&#x754C;&#x8BBF;&#x95EE;&#xFF08;&#x867D;&#x7136;vector&#x8D8A;&#x754C;&#x8BBF;&#x95EE;&#x4F3C;&#x4E4E;&#x6CA1;&#x6709;&#x9519;&#x8BEF;&#x3002;&#x3002;&#x3002;&#xFF09;
        if(pos == v.size()) cout << pos << endl;
        else cout << (v[pos].first==n?pos+1:pos) << endl;
    }
    return 0;
}

J Counting Triangles

题意

对一个有n个顶点的无向完全图，每一条边都被染成黑色或白色，求有多少个三角形(a, b, c) ( a<b<c )使得边ab，边bc，边ac的颜色一样。由于输入太多，使用以下代码来生成图：

namespace GenHelper
{
    unsigned z1,z2,z3,z4,b,u;
    unsigned get()
    {
        b=((z1<<6)^z1)>>13;
        z1=((z1&4294967294U)<<18)^b;
        b=((z2<<2)^z2)>>27;
        z2=((z2&4294967288U)<<2)^b;
        b=((z3<<13)^z3)>>21;
        z3=((z3&4294967280U)<<7)^b;
        b=((z4<<3)^z4)>>12;
        z4=((z4&4294967168U)<<13)^b;
        return (z1^z2^z3^z4);
    }
    bool read() {
      while (!u) u = get();
      bool res = u & 1;
      u >>= 1; return res;
    }
    void srand(int x)
    {
        z1=x;
        z2=(~x)^0x233333333U;
        z3=x^0x1234598766U;
        z4=(~x)+51;
          u = 0;
    }
}
using namespace GenHelper;
bool edge[8005][8005];
int main() {
  int n, seed;
  cin >> n >> seed;
  srand(seed);
  for (int i = 0; i < n; i++)
        for (int j = i + 1; j < n; j++)
            edge[j][i] = edge[i][j] = read();
     return 0;
}

n个顶点编号从0到n-1,n<=8000,seed<=1e9

题解

这里有一个比较有趣且明显的地方，那就是所有的三角形要么三条边颜色相同，要么只有一条边颜色与其余两条边不同，那么答案其实就是所有可能的三角形个数减去这些异色的三角形即可。

那么该如何统计异色三角形的数量？

注意到，一个三角形有三个角，一个角由一个顶点和两条边组成，那么异色三角形中就有两个异色边的角，又能够注意到，一个异边角只能去组成其中一个三角形，那么异色三角形的数量其实就是所有异色角的一半。

异色角可以统计一个顶点黑色边和白色边的数量，排列组合相乘即可。

using namespace GenHelper;
int s[8005];

int main() {
  int n, seed;
  cin >> n >> seed;
  srand(seed);
  for (int i = 0; i < n; i++)
        for (int j = i + 1; j < n; j++)
        {
            int k = read();
            if(k == 0) ++s[i], ++s[j];
        }
    if(n == 1 || n == 2){
        cout << 0 << endl;
        return 0;
    }
    long long ans = 0;
    //&#x767D;&#x8272;&#x8FB9;&#x4E58;&#x4E0A;&#x9ED1;&#x8272;&#x8FB9;&#xFF08;&#x8981;&#x6CE8;&#x610F;&#x9664;&#x53BB;&#x81EA;&#x73AF;&#x8FB9;&#xFF09;
    for(int i=0;i<n;i++) ans += s[i]*(n-s[i]-1);

    //&#x8BA1;&#x7B97;&#x6240;&#x6709;&#x53EF;&#x80FD;&#x7684;&#x4E09;&#x89D2;&#x5F62;&#x6570;&#x91CF;
    for(int i=1;i<=n-2;i++) res += (n-i)*(n-i-1)/2;
    cout << res-ans/2 << endl;
     return 0;
}
