约数、约数个数与欧拉函数

2021-12-10
算法 > 数学-算法ACM数学

约数个数定理

$ 对于一个正整数的质因数分解N = p_{1}^{c_1} p_{2}^{c_2} … p_{n}^{c_n} $

$ 其约数个数为f(N) = (c_1 + 1)(c_2 + 1) … (c_n + 1) $

$ 所有约数和为g(N) = (1 + p_1 + … + p_1^{c_1})(1 + p_2 + … + p_2^{c_2}) … (1 + p_n + … + p_n^{c_n}) $

几个约数相关的模板

求某个数的约数或约数个数

复杂度$ O(\sqrt{N}) $

vector<int > num;
int cnt = 0;
for(int i = 1; i <= sqrt(N); i ++) if(N % i == 0) {
    num.push_back(i);
    ++ cnt;
    if(N / i != i) num.push_back(N / i), ++ cnt;
}

求N!的约数个数

复杂度$ O(NlogN) $

get_prim(n);
for(int i = 1; i <= cnt; i ++) {
    for(ll j = prim[i]; n / j; j *= prim[i]) ocnt[i] += n / j;
}
ll res = 1;
for(int i = 1; i <= cnt; i ++) res = res * (ocnt[i] + 1) % p;

线性筛求约数个数

复杂度$ O(N) $

int ocnt[maxn], scnt[maxn]; //ocnt&#x5B58;&#x7EA6;&#x6570;&#x4E2A;&#x6570;&#xFF0C;scnt&#x5B58;&#x4E00;&#x4E2A;&#x6570;&#x6700;&#x5C0F;&#x8D28;&#x56E0;&#x5B50;&#x7684;&#x4E2A;&#x6570;
void fun(int n)
{
    ocnt[1]=1; scnt[1]=1;
    for(int i=2;i<=n;i++)
    {
        if(!vis[i])
        {
            prime[++ cnt]=i;
            scnt[i]=1;
            ocnt[i]=2;
        }
        for(int j = 1; i * prim[j] <= n; j ++)
        {
            vis[i * prime[j]]=1;
            if(i % prime[j] == 0)
            {
                scnt[i * prime[j]] = scnt[i] + 1;
                ocnt[i * prime[j]] = ocnt[i] / (scnt[i] + 1) * (scnt[i] + 2);
                break;
            }
            scnt[i * prime[j]] = 1;
            ocnt[i * prime[j]] = ocnt[i] * 2;
        }
    }
}

线性筛求约数和

同理，一样要保存最小质因子p的等比项数之和，以便更新i%prim[j]的情况，复杂度$ O(N) $

const int N=1e5+5;
bool vis[N];
int prim[N];
long long sp[N],ans[N];
int cnt = 0;
void initial(int n)
{
    ans[1]=1;
    for (int i = 2; i <= n ; i ++)
    {
        if (!vis[i])
        {
            prim[++ cnt] = i;
            sp[i] = i + 1;
            ans[i] = i + 1;
        }
        for (int j = 0 ; i * prim[j] <= n; j ++)
        {
            vis[i * prim[j]] = 1;
            if (i % prim[j] == 0)
            {
                sp[i * prim[j]] = sp[i] * prim[j] + 1;
                ans[i * prim[j]] = ans[i] / sp[i] * sp[i * prim[j]];
                break;
            }
            ans[i * prim[j]] = ans[i] * ans[prim[j]];
            sp[i * prim[j]] = 1 + prim[j];
        }
    }
}

约数与约数个数题目

轻拍牛头

题意：

给n个整数，A1，A2，… An，求对于每一个数Ai，求其他数中有几个数是它的约数。

1 <= n <= 1e5，1 <= Ai <= 1e6

题解：

注意到，Ai的最大值为1e6，那么对于每一个Ai，可以在1e6范围内枚举它的倍数，然后记录这1e6个数中每个数的约数有多少个即可。

代码：

#include<bits/stdc++.h>
using namespace std;
int n, a[100010], b[100010], c[1000010];

int main()
{
    scanf("%d", &n);
    for(int i = 1; i <= n; i ++) scanf("%d", a + i);
    // &#x4FDD;&#x5B58;&#x539F;&#x6570;&#x7EC4;
    memcpy(b, a, sizeof a);
    sort(a + 1, a + n + 1);
    a[n + 1] = a[n] + 1;
    for(int i = 1, front = 0; i <= n; i ++) {
        if(a[i] == a[i + 1]) continue ;
        // i - front&#x662F;a[i]&#x7684;&#x4E2A;&#x6570;&#x6709;&#x51E0;&#x4E2A;
        for(int j = 1; a[i] * j <= 1e6; j ++)
            c[a[i] * j] += i - front;
        front = i;
    }
    for(int i = 1; i <= n; i ++) printf("%d\n", c[b[i]] - 1);
    return 0;
}

樱花

题意：

给定一个正整数n，求有几个正整数对(x, y)满足$ \frac{1}{x} + \frac{1}{y} = \frac{1}{n!} $

1 <= n <= 1e6

题解：

给原式做一个简单变换：$ x = \frac{n!y}{y - n!} $

$ 令k = y - n!（k > 0），x = \frac{(n!)^2}{k} + n! $

因此答案就为$ (n!)^2 $的约数个数

代码：

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int maxn = 1e6 + 10, p = 1e9 + 7;
ll n, cont[maxn];
int vis[maxn], prim[maxn], cnt;

void get_prim(ll n)
{
    for(int i = 2; i <= n; i ++) {
        if(!vis[i]) prim[++ cnt] = i;
        for(int j = 1; i * prim[j] <= n; j ++) {
            vis[i * prim[j]] = 1;
            if(i % prim[j] == 0) break;
        }
    }
}

int main()
{
    scanf("%lld", &n);
    get_prim(n);
    
    for(int i = 1; i <= cnt; i ++) {
        for(ll j = prim[i]; n / j; j *= prim[i]) cont[i] += n / j;
    }
    
    ll res = 1;
    for(int i = 1; i <= cnt; i ++)
        res = res * (2 * cont[i] + 1) % p;
        
    printf("%lld\n", res);
    
    return 0;
}

反素数

题意：

给定一个N，求一个不超过N的最大的数x，满足所有小于x的数的约数个数都小于x的约数个数。

1 <= N <= 2e9

题解：

直觉上可以线性筛约数个数，但是N太大了，不好。

那么考虑一个数的质因数分解，可以发现，前十个质数相乘已经>2e9了，同时2^31>2e9，所以可以发现枚举前九个质数以及它们的指数，然后求出约数个数和该数的值，比较更新即可。

那么2e9中的数枚举其质因数分解，那么不久等于枚举2e9个数吗，实际上不是的，2e9中其实有很多数没有枚举，因为我们只操作前九个质数，操作后面的质数肯定没有前面的好（个数不会多，数反而更大）。

代码：

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll n, res = 1, resg = 1;
int prim[15] = {0, 2, 3, 5, 7, 11, 13, 17, 19, 23};

ll fastpow(ll a, ll b)
{
    ll res = 1;
    while(b) {
        if(b & 1) res *= a;
        b >>= 1;
        a *= a;
    }
    return res;
}

void dfs(int k, int mx, ll g, ll ans)
{
    if(k > 9) {
        if(g > resg) res = ans, resg = g;
        else if(g == resg) res = min(res, ans);
        return ;
    }
    
    for(int i = mx; i >= 0; i --)
        if(ans <= n / fastpow(prim[k], i))
            dfs(k + 1, i, g * (i + 1), ans * fastpow(prim[k], i));
}

int main()
{
    scanf("%lld", &n);
    dfs(1, 30, 1, 1);
    printf("%lld\n", res);
    return 0;
}

Hankson的趣味题

题意：

n组数据，每组给四个正整数，a0，a1，b0，b1，a0能被a1整除，b1能被b0整除，求有几个x满足，x与a0的最大公约数为a1，x与b0的最小公倍数为b1。

1 <= n <= 2000，1 <= a0,a1,b0,b1 <= 2e9

题解：

其实这里可以用到一个结论，在2e9内的所有数中，对于每一个数来说，其约数个数其实并不大，实际上，2e9以下的数中，约数个数最大的不过几千而已。

$ (x, a_0) = a_1，[x, b_0] = b_1 $

因此可以在分解质因数后，dfs枚举b1的约数x，求出所有满足上式的情况。

代码：

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int maxn = 1e5 + 10;
int n, vis[maxn], prim[maxn], pcnt, pp[15], cc[15], cnt, res;
ll a0, a1, b0, b1;

void get_prim(ll n)
{
    for(int i = 2; i <= n; i ++) {
        if(!vis[i]) prim[++ pcnt] = i;
        for(int j = 1; i * prim[j] <= n; j ++) {
            vis[i * prim[j]] = 1;
            if(i % prim[j] == 0) break;
        }
    }
}

ll gcd(ll a, ll b) { return b ? gcd(b, a%b) : a; }
ll lcm(ll a, ll b) { return a / gcd(a, b) * b; }

ll fastpow(ll a, ll b) { ll res = 1; while(b) { if(b&1) res*=a; b>>=1; a*=a; } return res; }

void dfs(int k, ll ans)
{
    if(k > cnt) {
        ll x = ans;
        if(gcd(max(x, a0), min(x, a0)) == a1 && lcm(max(x, b0), min(x, b0)) == b1) ++ res;
        return ;
    }
    
    for(int i = 0; i <= cc[k]; i ++) {
        //best method : no fastpow
        //dfs(k + 1, ans);
        //ans *= pp[k];
        dfs(k + 1, ans * fastpow(pp[k], i));
    }
}

int main()
{
    scanf("%d", &n);
    get_prim(50000);
    while(n --)
    {
        cnt = res = 0;
        memset(cc, 0, sizeof cc);  //error error
        
        scanf("%lld%lld%lld%lld", &a0, &a1, &b0, &b1);
        ll b = b1;
        //best method : i <= pcnt ==> 1ll * prim[i] * prim[i] <= b
        for(int i = 1; i <= pcnt ; i ++) {
            int t = prim[i];
            if(b % t == 0) pp[++ cnt] = t;
            while(b % t == 0) ++ cc[cnt], b /= t;
        }
        if(b != 1) pp[++ cnt] = b, cc[cnt] = 1;
        
        dfs(1, 1);
        
        //error
        
        printf("%d\n", res);
    }
    return 0;
}

欧拉函数

对于正整数n，欧拉函数为在小于n的正整数中与n互质的数的个数。

欧拉函数性质

$ 性质1：对于任意一个质数p，\varphi(p) = p - 1 $

$ 证明显然。 $

$ 性质2：当gcd(n, m) = 1时，\varphi(nm) = \varphi(n) \varphi(m) （积性函数）$

$ 证明：可由性质4原式证明。 $

$ 性质3：若n = p^k，其中p为质数，则\varphi(n) = p^k-p^{k - 1} = (p - 1)p^{k - 1} $

$ 证明：1到n中除了p的倍数都与p^k互质，由1到n中p的个数为p^k \div p个 $

$ 性质4： 分解n = p_1^{c_1} p_2^{c_2} … p_m^{c_m}，则\varphi(n) = n\prod \limits_{i=1}^{m}(1-\frac{1}{p_i}) $

$ 由性质3和质因数分解可知，\varphi(n) = \prod \limits $
