约数之和

原题

题意：

给定两个自然数A, B，求$ A^{B} $的约数之和，结果mod 9901。

0 <= A, B <= 5e7

题解：

根据数论的相关知识，某个数约数和公式如下：

$$ 若分解A = p_1^{a_1}p_2^{a_2}…p_m^{a_m} \\ 约数和为s = (1+p_1+p_1^{2}+…+p_1^{a_1})(1+p_2+p_2^{2}+…+p_2^{a_2})…(1+p_m+p_m^{2}+…+p_m^{a_m}) $$

对于每一个$ p_i $，可以运用等比数列求和后快速幂计算，但是涉及到逆元取模比较麻烦。

可以用分治法来求等比数列的取模和

$$ 1.当m为偶数时，s = 1+p+p^{2}+…+p^{m} = (1+p+…+p^{\frac{m}{2}})+p^{\frac{m}{2}+1}*(1+p+…+p^{\frac{m}{2}-1}) \\ 2.当m为奇数时，s = 1+p+p^{2}+…+p^{m} = (1+p+…+p^{\frac{m}{2}})+p^{\frac{m}{2}+1}*(1+p+…+p^{\frac{m}{2}}) $$

这样就分成了两部分来求，注意当m为偶数的时候，分成的两部分不一样，如何分开求的话在时间上并没有优化，所以可以把偶数减一之后变成奇数后再继续递归即可。

代码：

1.为了防止爆int，query中的质数pp传入的时候注意要先取模

#include<bits/stdc++.h>
using namespace std;
const int p = 9901;

int fpow(int a, int b) {
    int res = 1;
    while(b) {
        if(b & 1) res = res * a % p;
        b >>= 1;
        a = a * a % p;
    }
    return res;
}

int query(int pp, int aa)
{
    if(aa == 0) return 1;
    // if(aa&1) return (query(pp, aa/2) + fpow(pp, aa/2+1) * query(pp, aa/2)) % p;
    // return ( query(pp, aa/2) + fpow(pp, aa/2+1) * query(pp, aa/2-1) ) % p;
    if(aa&1) return query(pp, aa/2) * (1 + fpow(pp, aa/2+1)) % p;
    return  ( query(pp, aa - 1) + fpow(pp, aa) ) % p;
}

int main()
{
    int A, B;
    scanf("%d%d", &A, &B);
    int res = 1, len = sqrt(A);
    for(int i = 2; i <= len; i ++) if(A % i == 0) {
        int cnt = 0; while(A % i == 0) ++ cnt, A /= i;
        res = res * query(i % p, cnt * B) % p; // i % p error
    }
    if(A != 1) res = res * query(A % p, B) % p;
    printf("%d\n", A == 0 ? 0 : res);
    return 0;
}

分形之城

原题

题意：

题解：

图上的分治，注意坐标的变换到底是如何变换的即可

代码：

#include<bits/stdc++.h>
#define x first
#define y second
using namespace std;
typedef long long ll;
typedef pair<int, int> pii;
int T, n;
ll A, B;

pii query(int n, ll node)
{
    pii res;
    if(n == 1) {
        if(node == 0) res = {0, 0};
        if(node == 1) res = {0, 1};
        if(node == 2) res = {1, 1};
        if(node == 3) res = {1, 0};
    }
    else {
        ll t = 1ll << 2 * (n - 1);
        ll cell = node / t;
        ll vnod = node % t;
        t = 1ll << n;
        if(cell == 0) {
            res = query(n - 1, vnod);
            swap(res.x, res.y);
        }
        if(cell == 1) {
            res = query(n - 1, vnod);
            res.y += t / 2;
        }
        if(cell == 2) {
            res = query(n - 1, vnod);
            res.x += t / 2;
            res.y += t / 2;
        }
        if(cell == 3) {
            pii temp = query(n - 1, vnod);
            res.x = t - 1 - temp.y;
            res.y = t / 2 - 1 - temp.x;
        }
    }
    
    return res;
}

void solve()
{
    scanf("%d%lld%lld", &n, &A, &B);
    pii a = query(n, A - 1);
    pii b = query(n, B - 1);
    //sqrt in ver use double, can not int or long long
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    double len = sqrt(dx * dx + dy * dy) * 10;
    printf("%.0f\n", round(len));
}

int main()
{
    scanf("%d", &T);
    while(T --)
        solve();
    return 0;
}
