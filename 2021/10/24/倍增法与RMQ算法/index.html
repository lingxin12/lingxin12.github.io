倍增法与RMQ算法

2021-10-24- 2021-12-14
算法 > 基础-算法ACM

算法简介

目的

用于求区间最值问题（不修改）

模板

原题

#include<bits/stdc++.h>
using namespace std;
int n, m, a[200010], f[200010][18];

void init()
{
    int len = 1;
    for(int j = 0; j <= 17; j ++)
    {
        for(int i = 1; i + len - 1 <= n; i ++)
            if(j == 0) f[i][j] = a[i];
            else f[i][j] = max(f[i][j - 1], f[i + (len >> 1)][j - 1]);
        len <<= 1;
    }
}

int main()
{
    scanf("%d", &n);
    for(int i = 1; i <= n; i ++) scanf("%d", a + i);
    
    init();
    
    scanf("%d", &m);
    while(m --)
    {
        int l, r, j = 0, len = 1;
        scanf("%d%d", &l, &r);
        int k = log(r - l + 1) / log(2);
        printf("%d\n", max(f[l][k], f[r - (1 << k) + 1][k]));
    }
    
    return 0;
}
