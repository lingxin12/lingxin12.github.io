vector<int> sub(vector<int> &A, vector<int> &B)
{
    vector<int> C;
    for (int i = 0, t = 0; i < A.size(); i ++ )
    {
        t = A[i] - t;
        if (i < B.size()) t -= B[i];
        C.push_back((t + 10) % 10);
        if (t < 0) t = 1;
        else t = 0;
    }

    while (C.size() > 1 && C.back() == 0) C.pop_back();
    return C;
}

        C.push_back(t % 10);
        t /= 10;

}

                    }
                }
            }
        }
        return true;
    }

void kruskal()
{

                }
                
            }
        }
    }
    
}

void query(int x, int y, long long &d1, long long &d2)
{
    static int tdis[maxn<<1];

    if(dep[y] > dep[x]) swap(x, y);

        if(dep[fa[x][k]] >= dep[y]) {
            tdis[++ cnt] = dis1[x][k];
            tdis[++ cnt] = dis2[x][k];
            x = fa[x][k];
        }
    }
    
    if(x != y) {
        for(int k = 16; k >= 0; k --) {
            if(fa[x][k] != fa[y][k]) {
                
                tdis[++ cnt] = dis1[x][k];
                tdis[++ cnt] = dis2[x][k];
                tdis[++ cnt] = dis1[y][k];
                tdis[++ cnt] = dis2[y][k];
                
                x = fa[x][k];
                y = fa[y][k];
            }
        }
        
        tdis[++ cnt] = dis1[x][0];
        tdis[++ cnt] = dis1[y][0];
    }
    
    d1 = d2 = -linf;
    for(int i = 1; i <= cnt; i ++) {
        long long d = tdis[i];
        if(d > d1) d2 = d1, d1 = d;
        else if(d != d1 && d > d2) d2 = d;
    }
    
    if(d2 == -linf) d2 = linf;
}
