1.牛客50582 Fibonacci

https://ac.nowcoder.com/acm/problem/50582

题目：

斐波那契数列：f(0) = 0，f(1)=1，f(n)=f(n-1)+f(n-2)。

求f(n) mod $10^{4}$

分析：

矩阵快速幂。

代码：

#include<bits/stdc++.h>
#define enl puts("")
#define pii pair<int,int>
#define pdd pair<double , double >
#define pss pair<string , string >
#define pb push_back
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
const int mod = 1e4;
int a[3][3],n;

void mul(int a[3][3],int b[3][3])
{
    int c[3][3];
    memset(c,0,sizeof(c));
    for(int i=1;i<=2;i++)
        for(int j=1;j<=2;j++)
            for(int k=1;k<=2;k++)
                c[i][j] = ( c[i][j] + (a[i][k] * b[k][j]) % mod ) % mod;
    for(int i=1;i<=2;i++)
        for(int j=1;j<=2;j++)
            a[i][j] = c[i][j];
}

void fastpow(int a[3][3],int b)
{
    int c[3][3];
    for(int i=1;i<=2;i++) for(int j=1;j<=2;j++) c[i][j] = i==j?1:0;
    while(b) {
        if(b&1) mul(c,a);
        b >>= 1;
        mul(a,a);
    }
    for(int i=1;i<=2;i++) for(int j=1;j<=2;j++) a[i][j] = c[i][j];
}

int main()
{
    //freopen("data.in","r",stdin);
    while(~scanf("%d",&n) && n!=-1)
    {
        if(n == 0 || n == 1)
        {
            printf("%d\n",n==0?0:1);
            continue;
        }
        for(int i=1;i<=2;i++) for(int j=1;j<=2;j++) a[i][j] = 1;
        a[2][2] = 0;
        fastpow(a,n-1);
        printf("%d\n",a[1][1]);
    }
     return 0;
}

4.牛客14686 集合中的质数

https://ac.nowcoder.com/acm/problem/14686

题目：

集合里面有n个质数。

请你求出从 1 到 m 的所有数中，至少能被集合中的一个数整除的数的个数。

对于100%的数据，有n<=20，m为有符号64位正整数，集合内质数<=1000000000

分析：

对于每个质数，找到1到m中有没有被其整除的数，一共有几个。设a[i]表示第i个质数，明显有$\frac{m}{a[i]}$个。所以把所有质数对应的个数都加起来即可。

但是这时候发现有重复运算，例如3，5。15的倍数即是3的倍数也是5的倍数，有重复，得减去所有15的倍数。若是3，5，7，如果减去了3与5的倍数，5与7的倍数，3与7的倍数，这时候就会减多了，比如105的倍数是这3个数的倍数，105的倍数对答案的贡献第一次被加了三次，第二次又减了三次，所以这时候得加上3与5与7的倍数。

很明显可以发现，这就是容斥原理，具体可参考：

https://oi-wiki.org/math/inclusion-exclusion-principle/

对于选有顺序的排列，dfs即可。

代码：

#include<bits/stdc++.h>
#define enl puts("")
#define pii pair<int,int>
#define pdd pair<double , double >
#define pss pair<string , string >
#define pb push_back
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
ll n,m,a[30],b[30],res = 0;

void dfs(int tot,int k,int f,int start)
{
    if(tot == k) {
        ll ans = 1;
        for(int i=1;i<=n;i++) if(b[i] == 1) {
            //&#x7531;&#x4E8E;&#x8D28;&#x6570;&#x8FDE;&#x7EED;&#x76F8;&#x4E58;&#x4F1A;&#x6709;&#x6EA2;&#x51FA;LL&#x7684;&#x98CE;&#x9669;&#xFF0C;&#x53EF;&#x7528;&#x9664;&#x6CD5;+&#x7F29;&#x653E;&#xFF08;&#x52A0;&#x4E86;1&#xFF09;&#x5224;&#x65AD;&#x662F;&#x5426;&#x6EA2;&#x51FA;
            if(m/a[i]+1 < ans) return ;
            ans *= a[i];
        }
        res += f*(m/ans);
    }
    for(int i=start;i<=n;i++) {
        b[i] = 1;
        dfs(tot+1,k,f,i+1);
        b[i] = 0;
    }
}

int main()
{
    scanf("%lld%lld",&n,&m);
    for(int i=1;i<=n;i++) scanf("%lld",a+i);
    for(int i=1;i<=n;i++) res += m/a[i];
    int f = -1;
    sort(a+1,a+n+1);
    for(int i=2;i<=n;i++) {
        dfs(0,i,f,1);
        f = -f;
    }
    printf("%lld\n",res);
      return 0;
}

5.牛客15068 一个小问题

https://ac.nowcoder.com/acm/problem/15068

题目：

uu遇到了一个小问题，可是他不想答。你能替他解决这个问题吗？

问题：给你k对a和r是否存在一个正整数x使每队a和r都满足：x mod a=r，求最小正解x或无解。

k<=1e5，0<=r<a<1e5

分析：

中国剩余定理模板。

https://blog.csdn.net/ccnuacmhdu/article/details/77374224

拓展中国剩余定理：https://blog.csdn.net/qq_40772692/article/details/81872831

代码：

#include<bits/stdc++.h>
#define enl puts("")
#define pii pair<int,int>
#define pdd pair<double , double >
#define pss pair<string , string >
#define pb push_back
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
const int maxn = 1e5 + 2;
ll k,a[maxn],r[maxn];
int vis[maxn];
vector<ll > prim;

//&#x5224;&#x65AD;&#x4E24;&#x4E24;&#x4E92;&#x7D20;&#xFF0C;&#x7528;&#x5206;&#x89E3;&#x5B9A;&#x7406;
bool check()
{
    memset(vis,0,sizeof(vis));
    for(int i=1;i<=k;i++) {
        ll temp = a[i];
        ll maxx = sqrt(temp);
        for(int j=0;j<prim.size()&&prim[j]<=maxx;j++) if(temp%prim[j]==0) {
            if(vis[prim[j]]) return false;
            vis[prim[j]] = 1;
            while(temp%prim[j]==0) temp/=prim[j];
        }
        if(temp != 1) {
            if(vis[temp] == 1) return false;
            else vis[temp] = 1;
        }
    }
    return true;
}

void ex_gcd(ll a, ll b, ll& d, ll& x, ll& y) 
{ 
    if(!b) { 
        d = a; x = 1; y = 0; 
    } else { 
        ex_gcd(b, a%b, d, y, x); y -= x*(a/b); 
    } 
}

ll inv(ll a,ll p)  //&#x5982;&#x679C;&#x4E0D;&#x5B58;&#x5728;&#x8FD4;&#x56DE;-1
{
    ll d,x,y;
    ex_gcd(a,p,d,x,y);
    return d == 1 ? (x%p + p) % p : -1;  //x&#x53EF;&#x80FD;&#x4E3A;&#x8D1F;&#x6570;
}

int main()
{
    int nn = 1e5;
    int mm = sqrt(nn + 0.5);
    for(int i=2;i<=mm;i++) if(!vis[i])
        for(int j=i*i;j<=nn;j+=i) vis[i] = 1;
    for(int i=2;i<=nn;i++) if(!vis[i]) prim.pb(i);
    scanf("%lld",&k);
    for(int i=1;i<=k;i++) scanf("%lld%lld",a+i,r+i);
    if(!check()) printf("-1\n");
    else {
        //&#x4E2D;&#x56FD;&#x5269;&#x4F59;&#x5B9A;&#x7406;&#x6A21;&#x677F;&#xFF0C;A&#x5373;&#x4E3A;M&#xFF0C;a[i]&#x5373;&#x4E3A;mi&#xFF0C;ri&#x4E3A;ai
        ll x = 0, A = 1;
        for(int i=1;i<=k;i++) A *= a[i];
        for(int i=1;i<=k;i++) {
            ll Ai = A/a[i];
            x = (x + r[i] * Ai * inv(Ai,a[i]) % A) % A;
        }
        printf("%lld\n",x);
    }
      return 0;
}

8.牛客25729 Binary Number

https://ac.nowcoder.com/acm/problem/25729

题目：

给出n，m，k（1≤m≤n≤31，1≤k≤2×1e8），求对于一个n位二进制数（最高位必须为1，比如00110为3位二进制数）且有m个1，第k小的数是几，输出其二进制形式。（输入保证合法）

比如n=5，m=3，k=3。最小为10011，次小为10101，次次小为10110，故输出10110。

分析：

先默认最高位为1，问题化为在n-1位中填入m-1个1（其实就是排列组合），生成第k小即可。

举一个简单的例子：若n=7，m=4，k=7。

从小到大把前几个合法的数字写出来。（前面两位数忽略）

1000111，1001011，1001101，1001110，1010011，1010101，*(1010110)*，1011001， 1011010， 1011100……

问题化为求在后6个位置填上3个1时第k小，

根据最前面1所在的位置把排列好的数分类，明显发现， $\Large c_2^{2} + \Large c_3^{2}$ ≤ k=7 ≤ $\Large c_2^{2} + \Large c_3^{2} + \Large c_4^{2}$

所以再把一个1填在第5个位置上，然后问题又化为了在后4个位置上填上2个1，第$k - \Large c_2^{2} - \Large c_3^{2}$小，然后继续上述过程就可以了。

代码：

#include<bits/stdc++.h>
#define enl puts("")
#define pii pair<int,int>
#define pdd pair<double , double >
#define pss pair<string , string >
#define pb push_back
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
int k,C[50][50];
int n,m,t;

int main()
{
    //&#x9884;&#x5904;&#x7406;&#x7EC4;&#x5408;&#x6570;
    memset(C,0,sizeof(C));
    C[0][0] = 1;
    for(int i=1;i<=30;i++)
        for(int j=0;j<=i;j++)
            C[i][j] = j==0?1:C[i-1][j]+C[i-1][j-1];
      scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d%d",&n,&m,&k);
        printf("1");
        m-=1;
        int s = n-1, ans;
        //&#x5728;s&#x4F4D;&#x4E0A;&#x586B;m&#x4E2A;1&#xFF0C;ans&#x7528;&#x6765;&#x4FDD;&#x5B58;&#x591A;&#x4E2A;&#x7EC4;&#x5408;&#x6570;&#x76F8;&#x52A0;&#x7684;&#x524D;&#x7F00;&#x548C;
        while(m>0)
        {
            int i;
            ans = 0;
            //&#x627E;&#x5230;k&#x5904;&#x4E8E;&#x54EA;&#x4E2A;&#x533A;&#x95F4;
            for(i=m-1;i<=s;i++) {
                if(i == s || (ans < k && k <= ans+C[i][m-1])) break;
                ans += C[i][m-1];
            }
            k -= ans;
            //&#x5728;&#x7B2C;i+1&#x4F4D;&#x586B;&#x4E0A;1&#xFF0C;&#x524D;&#x9762;&#x90FD;&#x586B;&#x4E0A;0
            for(;s>i+1;s--) printf("0");
            printf("1");
            s-=1;
            m-=1;
        }
        //&#x6700;&#x540E;&#x628A;0&#x8865;&#x9F50;
        for(;s>0;s--) printf("0");
        enl;
    }
    return 0;
}

struct One
{
    ll a,b,p,x;

    One() {}

    {
        if(!b) { 
            d = a; x = 1; y = 0; 
        } else { 
            ex_gcd(b, a%b, d, y, x); y -= x*(a/b); 
        } 
    }

    {
        ll d,y;
        ex_gcd(a,p,d,x,y); //ax+py=b

            x*=b/d; y*=b/d;
            x = (x%p+p)%p;
        }
    }

        cin >> a >> p;
        b = 1;
        b = (b%p+p)%p;
        solve();
        if(x==-1) cout << "No solution!" << endl;
        else cout << x << endl;
    }
};

{
    One o;
    o.Main();
      return 0;
}
